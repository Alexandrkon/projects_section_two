145  // take the spreadsheet, and convert it to a bundle of
146  // conformance resources
147  public Bundle parse(FetchedFile f) throws Exception {
148    try {
149      name = f.getName();
150      bundle = new Bundle();
151      bundle.setType(BundleType.COLLECTION);
152      bundle.setId(name);
153      xls = new XLSXmlParser(new ByteArrayInputStream(f.getSource()), f.getName());
154      checkMappings();
155      loadBindings();
156      loadMetadata(f);
157      loadExtensions(f.getErrors());
158      List<String> namedSheets = new ArrayList<String>();
159      if (hasMetadata("title")) {
160        f.setTitle(metadata("title"));
161        if (hasMetadata("name"))
162          f.setName(metadata("name"));
163      } else {
164        if (hasMetadata("name")) {
165          f.setTitle(metadata("name"));
166          f.setName(metadata("name"));
167        }
168      }
169
170      StructureDefinition first = null;
171      if (hasMetadata("published.structure")) {
172        for (String n : metadata.get("published.structure")) {
173          if (!Utilities.noString(n)) {
174            supplementMappings(n);
175          }
176        }
177        for (String n : metadata.get("published.structure")) {
178          if (!Utilities.noString(n)) {
179            StructureDefinition sd = parseProfileSheet(n, namedSheets, f.getErrors(), false);
180            if (first == null)
181              first = sd;
182          }
183        }
184
185        int i = 0;
186        while (i < namedSheets.size()) {
187          parseProfileSheet(namedSheets.get(i), namedSheets, f.getErrors(), false);
188          i++;
189        }
190      } else if (!hasMetadata("extensions")) {
191        first = parseProfileSheet("Data Elements", namedSheets, f.getErrors(), true);
192      }
193      if (namedSheets.isEmpty() && xls.getSheets().containsKey("Search"))
194        readSearchParams(xls.getSheets().get("Search"), first);
195
196      if (xls.getSheets().containsKey("Operations"))
197        readOperations(loadSheet("Operations"));
198
199      if (first != null)
200        processMetadata(first);
201
202      checkOutputs(f);
203
204    } catch (Exception e) {
205      throw new Exception("exception parsing pack "+f.getName()+": "+e.getMessage(), e);
206    }
207
208    return bundle;
209  }

211  private void supplementMappings(String name) {
212    sheet = loadSheet(name);
213    if (sheet==null)
214      throw new Error("Unable to load sheet " + name);
215    List<String> mappingNames = sheet.getColumnNamesBySuffix(" Mapping");
216    List<String> loadedMappings = new ArrayList<String>();
217    for (MappingSpace m : mappings.values()) {
218      loadedMappings.add(m.getColumnName());
219    }
220    mappingNames.removeAll(loadedMappings);
221    int i = 0 - mappingNames.size();
222    for (String column : mappingNames) {
223      i++;
224      String mapname = column.substring(0, column.length() - 8);
225      MappingSpace m = new MappingSpace(column, mapname, mapname.toLowerCase().replace(' ',  '-'), i, true, false, false, null);
226      mappings.put("http://unknown.org/" + mapname, m);
227    }
228  }

259  private StructureDefinition parseProfileSheet(String n, List<String> namedSheets, List<ValidationMessage> issues, boolean logical) throws Exception {
260    StructureDefinition sd = new StructureDefinition();
261
262    Map<String, ElementDefinitionConstraintComponent> invariants = new HashMap<String, ElementDefinitionConstraintComponent>();
263    String name = logical ? "Invariants" : n+"-Inv";
264    sheet = loadSheet(name);
265    if (sheet != null)
266      invariants = readInvariants(sheet, n, name);
267
268    sheet = loadSheet(n);
269    if (sheet == null)
270      throw new Exception("The StructureDefinition referred to a tab by the name of '"+n+"', but no tab by the name could be found");
271    for (int row = 0; row < sheet.rows.size(); row++) {
272      ElementDefinition e = processLine(sd, sheet, row, invariants, true, row == 0);
273      if (e != null)
274        for (TypeRefComponent t : e.getType()) {
275          if (t.hasProfile() && !"Extension".equals(t.getWorkingCode()) && t.getProfile().get(0).getValue().startsWith("#")) {
276            if (!namedSheets.contains(t.getProfile().get(0).getValue().substring(1)))
277              namedSheets.add(t.getProfile().get(0).getValue().substring(1));
278          }
279        }
280    }
281
282    if (logical) {
283      sd.setKind(StructureDefinitionKind.LOGICAL);
284      sd.setId(sd.getDifferential().getElement().get(0).getPath());
285      sd.getDifferential().getElementFirstRep().getType().clear();
286      sd.setType(sd.getDifferential().getElementFirstRep().getPath());
287      sd.setBaseDefinition("http://hl7.org/fhir/StructureDefinition/Element");
288      sd.setDerivation(TypeDerivationRule.SPECIALIZATION);
289      sd.setAbstract(false);
290    } else {
291      sd.setKind(StructureDefinitionKind.RESOURCE);
292      sd.setDerivation(TypeDerivationRule.CONSTRAINT);
293      sd.setAbstract(false);
294      sd.setId(n.toLowerCase());
295      sd.setType(sd.getDifferential().getElementFirstRep().getPath());
296      if (sd.getDifferential().getElementFirstRep().hasSliceName()) {
297        sd.setName(sd.getDifferential().getElementFirstRep().getSliceName());
298        sd.getDifferential().getElementFirstRep().setSliceName(null);
299      }
300      if (sd.getDifferential().getElementFirstRep().getType().size() == 1 && sd.getDifferential().getElementFirstRep().getType().get(0).hasProfile())
301        sd.setBaseDefinition(sd.getDifferential().getElementFirstRep().getType().get(0).getProfile().get(0).getValue());
302      else
303        sd.setBaseDefinition("http://hl7.org/fhir/StructureDefinition/"+sd.getType());
304      if (!context.getResourceNames().contains(sd.getType()) && !context.getTypeNames().contains(sd.getType()))
305        throw new Exception("Unknown Resource "+sd.getType());
306    }
307    sd.getDifferential().getElementFirstRep().getType().clear();
308    sd.setUrl(base+"/StructureDefinition/"+sd.getId());
309    if ("http://hl7.org/fhir".equals(base))
310      sd.setVersion(Constants.VERSION);
311    bundle.addEntry().setResource(sd).setFullUrl(sd.getUrl());
312
313    // Changed the default from metadata to Short because the former caused problems when there are multiple sheets in a workbook
314    if (sheet.hasColumn(0, "Definition"))
315      sd.setDescription(sheet.getColumn(0, "Definition"));
316    if (sheet.hasColumn(0, "Profile.title"))
317      sd.setTitle(sheet.getColumn(0, "Profile.title"));
318    else if (sd.getDifferential().getElementFirstRep().hasShort())
319      sd.setTitle(sd.getDifferential().getElementFirstRep().getShort());
320    if (!sd.hasName()) {
321      if (!sd.hasName() && sd.getDifferential().getElementFirstRep().hasShort())
322        sd.setName(sd.getDifferential().getElementFirstRep().getShort());
323      else if (hasMetadata("name"))
324        sd.setName(metadata("name"));
325      else if (sheet.hasColumn(0, "Profile.name"))
326        sd.setName(sheet.getColumn(0, "Profile.name"));
327      else
328        sd.setName("UNKNOWN");
329    }
330    sheet = loadSheet(n + "-Extensions");
331    if (sheet != null) {
332      int row = 0;
333      while (row < sheet.rows.size()) {
334        if (sheet.getColumn(row, "Code").startsWith("!"))
335          row++;
336        else
337          row = processExtension(sheet, row, metadata("extension.uri"), issues, invariants);
338      }
339    }
340
341    sheet = loadSheet(n+"-Search");
342    if (sheet != null) {
343      readSearchParams(sd, sheet, true);
344    }
345
346    if (invariants != null) {
347      for (ElementDefinitionConstraintComponent inv : invariants.values()) {
348        if (Utilities.noString(inv.getUserString("context")))
349          throw new Exception("Profile "+sd.getId()+" Invariant "+inv.getId()+" has no context");
350        else {
351          ElementDefinition ed = findContext(sd, inv.getUserString("context"), "Profile "+sd.getId()+" Invariant "+inv.getId()+" Context");
352          ed.getConstraint().add(inv);
353          if (Utilities.noString(inv.getXpath())) {
354            throw new Exception("Profile "+sd.getId()+" Invariant "+inv.getId()+" ("+inv.getHuman()+") has no XPath statement");
355          }
356          if (Utilities.noString(inv.getExpression())) {
357            throw new Exception("Profile "+sd.getId()+" Invariant "+inv.getId()+" ("+inv.getHuman()+") has no Expression statement");
358          }
359          else if (inv.getXpath().contains("\""))
360            throw new Exception("Profile "+sd.getId()+" Invariant "+inv.getId()+" ("+inv.getHuman()+") contains a \" character: "+inv.getXpath());
361        }
362      }
363    }
364
365    sd.setPublisher(metadata("author.name"));
366    if (Utilities.noString(metadata("experimental")))
367      sd.setExperimental("true".equals(metadata("experimental")));
368    if (hasMetadata("author.reference"))
369      sd.addContact().getTelecom().add(Factory.newContactPoint(ContactPointSystem.URL, metadata("author.reference")));
370    if (hasMetadata("date"))
371      sd.setDateElement(Factory.newDateTime(metadata("date").substring(0, 10)));
372    else
373      sd.setDate(genDate.getTime());
374      if (hasMetadata("description"))
375        sd.setDescription(metadata("description"));
376    if (hasMetadata("version"))
377      sd.setVersion(metadata("version"));
378    if (hasMetadata("status"))
379      sd.setStatus(PublicationStatus.fromCode(metadata("status")));
380    else
381      sd.setStatus(PublicationStatus.DRAFT);
382
383    ProfileUtilities utils = new ProfileUtilities(this.context, issues, null);
384    utils.setIds(sd, false);
385    return sd;
386  }

388  private ElementDefinition findContext(StructureDefinition sd, String context, String message) throws Exception {
389    for (ElementDefinition ed : sd.getDifferential().getElement())
390      if (ed.getPath().equals(context))
391        return ed;
392    throw new Exception("No Context found for "+context+" at "+message);
393  }

395  private void readSearchParams(Sheet sheet, StructureDefinition sd) throws Exception {
396    if (sheet != null) {
397      for (int row = 0; row < sheet.rows.size(); row++) {
398        if (!sheet.hasColumn(row, "Name"))
399          throw new Exception("Search Param has no name "+ getLocation(row));
400        String n = sheet.getColumn(row, "Name");
401        if (!n.startsWith("!")) {
402          SearchParameter sp = new SearchParameter();
403          sp.setId(sd.getId()+"-"+n);
404          sp.setName("Search Parameter "+n);
405          sp.setUrl(base+"/SearchParameter/"+sp.getId());
406          sp.setStatus(sd.getStatus());
407          sp.setExperimental(sd.getExperimental());
408          if (!sheet.hasColumn(row, "Type"))
409            throw new Exception("Search Param "+sd.getId()+"-"+n+" has no type "+ getLocation(row));
410          sp.setType(readSearchType(sheet.getColumn(row, "Type"), row));
411          sp.setDescription(sheet.getColumn(row, "Description"));
412          sp.setXpathUsage(readSearchXPathUsage(sheet.getColumn(row, "Expression Usage"), row));
413          sp.setXpath(sheet.getColumn(row, "XPath"));
414          sp.setExpression(sheet.getColumn(row, "Expression"));
415          if (!sp.hasExpression())
416            sp.setExpression(sheet.getColumn(row, "Path"));
417          if (!sp.hasDescription())
418            throw new Exception("Search Param "+sd.getId()+"/"+n+" has no description "+ getLocation(row));
419//          FHIRPathEngine engine = new FHIRPathEngine(context);
420//          engine.check(null, sd.getType(), sd.getType(), sp.getExpression());
421//          bundle.addEntry().setResource(sp).setFullUrl(sp.getUrl());
422        }
423      }
424    }
425  }

452  private void loadMetadata(FetchedFile f) throws Exception {
453    sheet = loadSheet("Metadata");
454    if (sheet != null) {
455      for (int row = 0; row < sheet.rows.size(); row++) {
456        String n = sheet.getColumn(row, "Name");
457        String v = sheet.getColumn(row, "Value");
458        if (n != null && v != null) {
459          if (metadata.containsKey(n))
460            metadata.get(n).add(v);
461          else {
462            ArrayList<String> vl = new ArrayList<String>();
463            vl.add(v);
464            metadata.put(n, vl);
465          }
466        }
467      }
468    }
469    if (!hasMetadata("extension.uri") || !metadata("extension.uri").startsWith(base))
470      message(f, "extension.uri must be defined for IG spreadsheets, and must start with "+base, null, IssueType.BUSINESSRULE, IssueSeverity.ERROR);
471  }

487    }

490  private void readBindings(Sheet sheet) throws Exception {
491    for (int row = 0; row < sheet.rows.size(); row++) {
492      String bindingName = sheet.getColumn(row, "Binding Name");
493
494      // Ignore bindings whose name start with "!"
495      if (Utilities.noString(bindingName) || bindingName.startsWith("!")) continue;
496
497      ElementDefinitionBindingComponent bs = new ElementDefinitionBindingComponent();
498      bindings.put(bindingName, bs);
499      bs.setDescription(sheet.getColumn(row, "Definition"));
500      bs.setStrength(readBindingStrength(sheet.getColumn(row, "Conformance"), row));
501
502      if (sheet.hasColumn("max-valueset") && !Utilities.noString(sheet.getColumn(row, "max-valueset"))) {
503        bs.addExtension().setUrl("http://hl7.org/fhir/StructureDefinition/elementdefinition-maxValueSet").setValue(new CanonicalType(sheet.getColumn(row, "max-valueset")));
504      }
505
506      if (sheet.hasColumn("min-valueset") && !Utilities.noString(sheet.getColumn(row, "min-valueset"))) {
507                bs.addExtension().setUrl("http://hl7.org/fhir/StructureDefinition/elementdefinition-minValueSet").setValue(new CanonicalType(sheet.getColumn(row, "min-valueset")));
508      }
509
510      String type = sheet.getColumn(row, "Binding");
511      if (type == null || "".equals(type) || "unbound".equals(type)) {
512        // nothing
513      } else if (type.equals("code list")) {
514//        throw new Error("Code list is not yet supported in "+ getLocation(row));
515        String ref = sheet.getColumn(row, "Reference");
516
517        ValueSet vs = ValueSetUtilities.makeShareable(new ValueSet());
518        vs.setId(ref.substring(1));
519        vs.setUrl(base+"/ValueSet/"+ref.substring(1));
520        if (sheet.hasColumn("Version"))
521          vs.setVersion(sheet.getColumn(row, "Version"));
522        else
523          vs.setVersion(metadata("version"));
524        if (Utilities.noString(metadata("experimental")))
525          vs.setExperimental("true".equals(metadata("experimental")));
526        vs.setPublisher(metadata("author.name"));
527        bs.setValueSet(vs.getUrl()+"|");
528        bundle.addEntry().setResource(vs).setFullUrl(vs.getUrl());
529        vs.setName(bindingName);

531        if (!Utilities.noString(oid))
532          ValueSetUtilities.setOID(vs, oid);
533        String st = sheet.getColumn(row, "Status");
534        if (Utilities.noString(st))
535          st = "draft";

540        String em = sheet.getColumn(row, "Website");
541        if (em != null)
542          vs.getContactFirstRep().addTelecom().setSystem(ContactPointSystem.EMAIL).setValue(em);
543        vs.setCopyright(sheet.getColumn(row, "Copyright"));
544        vs.setDescription(sheet.getColumn(row, "Definition"));
545        Sheet css = xls.getSheets().get(ref.substring(1));
546        if (css == null)
547          throw new Exception("Error parsing binding "+bindingName+": code list reference '"+ref+"' not resolved");
548        loadValueSet(vs, css, ref.substring(1));
549      } else if (type.equals("special")) {
550        throw new Error("Binding type Special is not allowed in implementation guides"+ getLocation(row));
551      } else if (type.equals("reference")) {
552        throw new Exception("The binding type 'reference' is no longer supported");
553      } else if (type.equals("value set")) {
554        String ref = sheet.getColumn(row, "Reference");
555        String id = ref.startsWith("valueset-") ? ref.substring(9) : ref;
556        if (!ref.startsWith("http:") && !ref.startsWith("https:") && !ref.startsWith("ValueSet/")) {
557          valuesetsToLoad.put(id, ref);
558          ref = Utilities.pathURL(base, "ValueSet", id);
559        }
560        bs.setValueSet(ref);
561      } else {
562        throw new Exception("Unknown Binding: "+type+ getLocation(row));
563      }
564    }

703  private ElementDefinition processLine(StructureDefinition sd, Sheet sheet, int row, Map<String, ElementDefinitionConstraintComponent> invariants, boolean profile, boolean firstTime) throws Exception {
704    String path = sheet.getColumn(row, "Element");
705
706    if (path.startsWith("!"))
707      return null;
708    if (Utilities.noString(path))
709      throw new Exception("Error reading definitions - no path found @ " + getLocation(row));
710
711    if (path.contains("#"))
712      throw new Exception("Old path style @ " + getLocation(row));
713
714    String profileName = sheet.getColumn(row, "Profile Name");
715    String discriminator = sheet.getColumn(row, "Discriminator");
716
717    boolean isRoot = !path.contains(".");
718
719    ElementDefinition e = sd.getDifferential().addElement();
720    if (isRoot) {
721      e.setPath(path);
722    } else {
723      String arc = path.substring(0, path.lastIndexOf("."));
724      String leaf = path.substring(path.lastIndexOf(".")+1);
725      if (leaf.startsWith("@")) {
726        leaf = leaf.substring(1);
727        e.addRepresentation(PropertyRepresentation.XMLATTR);
728      }
729      e.setPath(arc+"."+leaf);
730    }
731    String c = sheet.getColumn(row, "Card.");
732    if (c == null || c.equals("") || c.startsWith("!")) {
733    } else {
734      String[] card = c.split("\\.\\.");
735      if (card.length != 2 || !Utilities.isInteger(card[0]) || (!"*".equals(card[1]) && !Utilities.isInteger(card[1])))
736        throw new Exception("Unable to parse Cardinality '" + c + "' " + c + " in " + getLocation(row) + " on " + path);
737      e.setMin(Integer.parseInt(card[0]));
738      e.setMax("*".equals(card[1]) ? "*" : card[1]);
739    }
740    e.setSliceName(profileName);
741    if (!Utilities.noString(discriminator)) {
742      e.getSlicing().setRules(SlicingRules.OPEN);
743      if (discriminator.contains("|")) {
744        throw new Error("We don't yet support ordered or non-open slicing when defining profiles with spreadsheets: " + discriminator);
745      }
746      for (String d : discriminator.split("\\,"))
747        if (!Utilities.noString(d))
748          e.getSlicing().addDiscriminator(ProfileUtilities.interpretR2Discriminator(d.trim(), false));
749      if (e.hasSliceName()) {
750        throw new Error("Slice name '"+profileName+"' on a root which has slicing: "+e.getPath());
751      }
752    }
753    doAliases(sheet, row, e);
754
755    if (sheet.hasColumn(row, "Is Modifier"))
756      e.setIsModifier(parseBoolean(sheet.getColumn(row, "Is Modifier"), row, false));
757    if (e.getIsModifier()) {
758      String reason = sheet.getColumn(row, "Modifier Reason");
759      if (Utilities.noString(reason)) {
760        if (path.endsWith(".modifierExtension")) {
761          reason = "Modifier extensions are expected to modify the meaning or interpretation of the element that contains them";
762        } else {
763          System.out.println("Missing IsModifierReason on "+path);
764          reason = "Not known why this is labelled a modifier";
765        }
766      }
767      e.setIsModifierReason(reason);
768    } else if (path.endsWith(".modifierExtension")) {
769      e.setIsModifier(true);
770      e.setIsModifierReason("Modifier extensions are expected to modify the meaning or interpretation of the element that contains them");
771    }
772    // later, this will get hooked in from the underlying definitions, but we need to know this now to validate the extension modifier matching
773    if (e.getPath().endsWith(".modifierExtension"))
774      e.setIsModifier(true);
775    e.setMustSupport(parseBoolean(sheet.getColumn(row, "Must Support"), row, false));
776
777    if (sheet.hasColumn(row, "Summary"))
778      e.setIsSummary(parseBoolean(sheet.getColumn(row, "Summary"), row, false));
779
780    String uml = sheet.getColumn(row, "UML");
781    if (!Utilities.noString(uml)) {
782      if (uml.contains(";")) {
783        String[] parts = uml.split("\\;");
784        e.setUserData("SvgLeft", parts[0]);
785        e.setUserData("SvgTop", parts[1]);
786        if (parts.length > 2)
787          e.setUserData("SvgWidth", parts[2]);
788      } else if (uml.startsWith("break:")) {
789        e.setUserData("UmlBreak", true);
790        e.setUserData("UmlDir", uml.substring(6));
791      } else {
792        e.setUserData("UmlDir", uml);
793      }
794    }
795    String s = sheet.getColumn(row, "Condition");
796    if (s != null && !s.equals(""))
797      throw new Exception("Found Condition in spreadsheet "+ getLocation(row));
798    s = sheet.getColumn(row, "Inv.");
799    if (s != null && !s.equals("")) {
800      for (String sn : s.split(",")) {
801        ElementDefinitionConstraintComponent inv = invariants.get(sn);
802        if (inv == null)
803          throw new Exception("unable to find Invariant '" + sn + "' "   + getLocation(row));
804        e.addCondition(inv.getKey());
805      }
806    }
807
808    if (!Utilities.noString(sheet.getColumn(row, "Type"))) {
809      TypeParser tp = new TypeParser();
810      List<TypeRef> types = tp.parse(sheet.getColumn(row, "Type"), true, metadata("extension.uri"), context, !path.contains("."));
811      if (types.size() == 1 && types.get(0).getName().startsWith("@"))
812        e.setContentReference("#"+types.get(0).getName().substring(1));
813      else if (types.size() > 0)
814        e.getType().addAll(tp.convert(context, e.getPath(), types, true, e));
815    }
816    String regex = sheet.getColumn(row, "Regex");
817    if (!Utilities.noString(regex) && e.hasType())
818      ToolingExtensions.addStringExtension(e.getType().get(0), ToolingExtensions.EXT_REGEX, regex);
819
820    if ((path.endsWith(".extension") || path.endsWith(".modifierExtension")) && e.hasType() && e.getType().get(0).hasProfile() && Utilities.noString(profileName))
821        throw new Exception("need to have a profile name if a profiled extension is referenced for "+ e.getType().get(0).getProfile());
822
823    String bindingName = sheet.getColumn(row, "Binding");
824    if (!Utilities.noString(bindingName)) {
825      ElementDefinitionBindingComponent binding = bindings.get(bindingName);
826      if (binding == null && !bindingName.startsWith("!"))
827        throw new Exception("Binding name "+bindingName+" could not be resolved in local spreadsheet");
828      e.setBinding(binding);
829    }
830    e.setShort(sheet.getColumn(row, "Short Name"));
831
832
833    e.setDefinition(Utilities.appendPeriod(processDefinition(sheet.getColumn(row, "Definition"))));
834
835    if (!Utilities.noString(sheet.getColumn(row, "Max Length")))
836      e.setMaxLength(Integer.parseInt(sheet.getColumn(row, "Max Length")));
837    e.setRequirements(Utilities.appendPeriod(sheet.getColumn(row, "Requirements")));
838    if (e.hasRequirements() && !e.getPath().contains(".")) {
839      sd.setPurpose(e.getRequirements());
840      e.setRequirements(null);
841    }
842    e.setComment(Utilities.appendPeriod(sheet.getColumn(row, "Comments")));
843    for (String n : mappings.keySet()) {
844      MappingSpace m = mappings.get(n);
845      String sm = sheet.getColumn(row, mappings.get(n).getColumnName());
846      if (!Utilities.noString(sm)) {
847        ElementDefinitionMappingComponent map = e.addMapping();
848        map.setIdentity(m.getId());
849        map.setMap(sm);
850        boolean found = false;
851        for (StructureDefinitionMappingComponent mm : sd.getMapping()) {
852          if (mm.getIdentity().equals(m.getId()))
853            found = true;
854        }
855        if (!found) {
856          StructureDefinitionMappingComponent mm = sd.addMapping();
857          mm.setIdentity(m.getId());
858          mm.setName(m.getTitle());
859          mm.setUri(n);
860        }
861      }
862    }
863    if (!Utilities.noString(sheet.getColumn(row, "Example")))
864      e.addExample().setLabel("General").setValue(processValue(sheet, row, "Example", sheet.getColumn(row, "Example"), e));
865    processOtherExamples(e, sheet, row);
866    String dh = sheet.getColumn(row, "Display Hint");
867    if (!Utilities.noString(dh))
868      ToolingExtensions.addDisplayHint(e, dh);
869    e.setFixed(processValue(sheet, row, "Value", sheet.getColumn(row, "Value"), e));
870    e.setPattern(processValue(sheet, row, "Pattern", sheet.getColumn(row, "Pattern"), e));
871    return e;
872  }

893  private DataType processStringToType(String type, String s, String path) throws Exception {
894    s = s.trim();
895    if (s.equalsIgnoreCase("Not Stated") || s.equalsIgnoreCase("n/a") || s.equalsIgnoreCase("-"))
896      return null;
897    if (Utilities.noString(type))
898      return new StringType(s);
899    if (type.equals("Reference")) {
900      return new Reference().setReference(s);
901    }
902    if (type.equals("Quantity")) {
903      int j = s.charAt(0) == '>' || s.charAt(0) == '<' ? 1 : 0;
904      int i = j;
905      while (i < s.length() && (Character.isDigit(s.charAt(i)) || s.charAt(i) == '.'))
906        i++;
907      if (i == j)
908        throw new Exception("Error parsing quantity value '"+s+"': must have the format [d][u] e.g. 50mm on "+path);
909      Quantity q = new Quantity();
910      q.setValue(new BigDecimal(s.substring(j, i)));
911      if (i < s.length()) {
912        q.setUnit(s.substring(i).trim());
913        q.setCode(s.substring(i).trim());
914        q.setSystem("http://unitsofmeasure.org");
915      }
916      if (j > 0)
917        q.setComparator(QuantityComparator.fromCode(s.substring(0, j)));
918      return q;
919    }
920    return new StringType(s);
921  }

955  private DataType processValue(Sheet sheet, int row, String column, String source, ElementDefinition e) throws Exception {
956    if (Utilities.noString(source))
957      return null;
958    if (e.getType().size() != 1)
959      throw new Exception("Unable to process "+column+" unless a single type is specified @ "+getLocation(row)+", type = \""+e.typeSummary()+"\", column = "+column);
960    String type = e.getType().get(0).getWorkingCode();
961    StructureDefinition sd = context.fetchTypeDefinition(type);
962    if (sd != null && sd.hasBaseDefinition() && sd.getDerivation() == TypeDerivationRule.CONSTRAINT)
963      type = sd.getType();
964
965    if (source.startsWith("{")) {
966      JsonParser json = new JsonParser();
967      try {
968        return json.parseType(source, type);
969      } catch (Exception e2) {
970        throw new Exception("Unable to parse json string: " + source+" as "+type+" because "+e2.getMessage(), e2);
971      }
972    } else if (source.startsWith("<")) {
973      XmlParser xml = new XmlParser();
974      try {
975        return xml.parseType(source, type);
976      } catch (Exception e2) {
977        throw new Exception("Unable to parse xml string: " + source+" as "+type+" because "+e2.getMessage(), e2);
978      }
979    } else {
980      source = source.trim();
981      if (source.startsWith("\"") && source.endsWith("\""))
982        source = source.substring(1, source.length()-1);
983
984      if (type.equals("string"))
985        return new StringType(source);
986      if (type.equals("boolean"))
987        return new BooleanType(Boolean.valueOf(source));
988      if (type.equals("integer"))
989        return new IntegerType(Integer.valueOf(source));
990      if (type.equals("unsignedInt"))
991        return new UnsignedIntType(Integer.valueOf(source));
992      if (type.equals("positiveInt"))
993        return new PositiveIntType(Integer.valueOf(source));
994      if (type.equals("decimal"))
995        return new DecimalType(new BigDecimal(source));
996      if (type.equals("base64Binary"))
997        return new Base64BinaryType(source);
998      if (type.equals("instant"))
999        return new InstantType(source);
1000      if (type.equals("uri"))
1001        return new UriType(source);
1002      if (type.equals("url"))
1003        return new UrlType(source);
1004      if (type.equals("canonical"))
1005        return new CanonicalType(source);
1006      if (type.equals("date"))
1007        return new DateType(source);
1008      if (type.equals("dateTime"))
1009        return new DateTimeType(source);
1010      if (type.equals("time"))
1011        return new TimeType(source);
1012      if (type.equals("code"))
1013        return new CodeType(source);
1014      if (type.equals("oid"))
1015        return new OidType(source);
1016      if (type.equals("uuid"))
1017        return new UuidType(source);
1018      if (type.equals("id"))
1019        return new IdType(source);
1020      if (type.startsWith("Reference(")) {
1021        Reference r = new Reference();
1022        r.setReference(source);
1023        return r;
1024      }
1025      if (type.equals("Period")) {
1026        if (source.contains("->")) {
1027          String[] parts = source.split("\\-\\>");
1028          Period p = new Period();
1029          p.setStartElement(new DateTimeType(parts[0].trim()));
1030          if (parts.length > 1)
1031            p.setEndElement(new DateTimeType(parts[1].trim()));
1032          return p;
1033
1034        } else
1035          throw new Exception("format not understood parsing "+source+" into a period");
1036      }
1037      if (type.equals("CodeableConcept")) {
1038        CodeableConcept cc = new CodeableConcept();
1039        if (source.contains(":")) {
1040          String[] parts = source.split("\\:");
1041          String system = "";
1042          if (parts[0].equalsIgnoreCase("SCT"))
1043            system = "http://snomed.info/sct";
1044          else if (parts[0].equalsIgnoreCase("LOINC"))
1045            system = "http://loinc.org";
1046          else if (parts[0].equalsIgnoreCase("AMTv2"))
1047            system = "http://nehta.gov.au/amtv2";
1048          else
1049            system = "http://hl7.org/fhir/"+parts[0];
1050          String code = parts[1];
1051          String display = parts.length > 2 ? parts[2] : null;
1052          cc.addCoding().setSystem(system).setCode(code).setDisplay(display);
1053        } else
1054          throw new Exception("format not understood parsing "+source+" into a codeable concept");
1055        return cc;
1056      }
1057      if (type.equals("Identifier")) {
1058        Identifier id = new Identifier();
1059        id.setSystem("urn:ietf:rfc:3986");
1060        id.setValue(source);
1061        return id;
1062      }
1063      if (type.equals("Quantity")) {
1064        int s = 0;
1065        if (source.startsWith("<=") || source.startsWith("=>"))
1066          s = 2;
1067        else if (source.startsWith("<") || source.startsWith(">"))
1068          s = 1;
1069        int i = s;
1070        while (i < source.length() && Character.isDigit(source.charAt(i)))
1071          i++;
1072        Quantity q = new Quantity();
1073        if (s > 0)
1074          q.setComparator(QuantityComparator.fromCode(source.substring(0, s)));
1075        if (i > s)
1076          q.setValue(new BigDecimal(source.substring(s, i)));
1077        if (i < source.length())
1078          q.setUnit(source.substring(i).trim());
1079        return q;
1080      }
1081
1082      throw new Exception("Unable to process primitive value '"+source+"' provided for "+column+" - unhandled type "+type+" @ " +getLocation(row));
1083    }
1084  }

1442    Map<String, OperationDefinition> ops = new HashMap<String, OperationDefinition>();
1443    Map<String, OperationDefinitionParameterComponent> params = new HashMap<String, OperationDefinitionParameterComponent>();

1453          if (!name.contains(".")) {

1457              throw new Exception("Invalid name on operation "+name+" - must be all lower case (use dashes) at " +getLocation(row));

1468              else if ("resource".equalsIgnoreCase(c))
1469                istype = true;
1470              else if ("instance".equalsIgnoreCase(c))
1471                instance = true;
1472              else

1474            }

1477            op.setUrl(base+"/OperationDefinition/"+name);
