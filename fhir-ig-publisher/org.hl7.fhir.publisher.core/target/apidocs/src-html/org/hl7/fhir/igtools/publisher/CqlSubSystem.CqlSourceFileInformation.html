103  /**
104   * Provides a library source provider that can resolve CQL library source from an Npm package
105   */
106  public class NpmLibrarySourceProvider implements LibrarySourceProvider {
107
108    @Override
109    public InputStream getLibrarySource(VersionedIdentifier identifier) {
110      // VersionedIdentifier.id: Name of the library
111      // VersionedIdentifier.system: Namespace for the library, as a URL
112      // VersionedIdentifier.version: Version of the library
113      for (NpmPackage p : packages) {
114        try {
115          InputStream s = p.loadByCanonicalVersion(identifier.getSystem()+"/Library/"+identifier.getId(), identifier.getVersion());
116          if (s != null) {
117            Library l = reader.readLibrary(s);
118            for (org.hl7.fhir.r5.model.Attachment a : l.getContent()) {
119              if (a.getContentType() != null && a.getContentType().equals("text/cql")) {
120                return new ByteArrayInputStream(a.getData());
121              }
122            }
123          }
124        } catch (IOException e) {
125          logger.logDebugMessage(ILoggingService.LogCategory.PROGRESS, String.format("Exceptions occurred attempting to load npm library source for %s", identifier.toString()));
126        }
127      }
128
129      return null;
130    }
131  }

209  /**
210   * Do the compile. Do not return any exceptions related to content; only thros exceptions for infrastructural issues 
211   * 
212   * note that it's not an error if there's no .cql files - this is called without checking for their existence
213   *  
214   * Any exception will stop the build cold.
215   */
216  public void execute() throws FHIRException {
217    try {
218      logger.logMessage("Translating CQL source");
219      fileMap = new HashMap<>();
220
221       // foreach folder
222      for (String folder : folders) {
223        translateFolder(folder);
224      }
225    }
226    catch (Exception E) {
227      logger.logDebugMessage(ILoggingService.LogCategory.PROGRESS, String.format("Errors occurred attempting to translate CQL content: %s", E.getMessage()));
228    }
229  }

231  /**
232   * Return CqlSourceFileInformation for the given filename
233   * @param filename Fully qualified name of the source file
234   * @return
235   */
236  public CqlSourceFileInformation getFileInformation(String filename) {
237    if (fileMap == null) {
238      throw new IllegalStateException("CQL File map is not available, execute has not been called");
239    }
240
241    if (!fileMap.containsKey(filename)) {
242      return null;
243    }
244
245    return this.fileMap.remove(filename);
246  }

249   * Called at the end after all getFileInformation have been called
250   * return any errors that didn't have any particular home, and also
251   * errors for any files that were linked but haven't been accessed using
252   * getFileInformation - these have been omitted from the IG, and that's

257  public List<ValidationMessage> getGeneralErrors() {
258    List<ValidationMessage> result = new ArrayList<>();
259
260    if (fileMap != null) {
261      for (Map.Entry<String, CqlSourceFileInformation> entry : fileMap.entrySet()) {
262        result.add(new ValidationMessage(ValidationMessage.Source.Publisher, ValidationMessage.IssueType.PROCESSING, entry.getKey(), "CQL source was not associated with a library resource in the IG.", ValidationMessage.IssueSeverity.ERROR));
263      }
264    }
265
266    return result;
267  }

269  /**
270   * Reads configuration file named cql-options.json from the given folder if present. Otherwise returns default options.
271   * @param folder
272   * @return
273   */
274  private CqlTranslatorOptions getTranslatorOptions(String folder) {
275    String optionsFileName = folder + File.separator + "cql-options.json";
276    CqlTranslatorOptions options = null;
277    File file = new File(optionsFileName);
278    if (file.exists()) {
279      options = CqlTranslatorOptionsMapper.fromFile(file.getAbsolutePath());
280    }
281    else {
282      options = CqlTranslatorOptions.defaultOptions();
283      if (!options.getFormats().contains(CqlTranslator.Format.XML)) {
284        options.getFormats().add(CqlTranslator.Format.XML);
285      }
286    }
287
288    return options;
289  }

291  private void translateFolder(String folder) {
292    logger.logMessage(String.format("Translating CQL source in folder %s", folder));
293
294    CqlTranslatorOptions options = getTranslatorOptions(folder);
295
296    // Setup
297    // Construct DefaultLibrarySourceProvider
298    // Construct FhirLibrarySourceProvider
299    ModelManager modelManager = new ModelManager();
300    LibraryManager libraryManager = new LibraryManager(modelManager);
301    libraryManager.getLibrarySourceLoader().registerProvider(new NpmLibrarySourceProvider());
302    libraryManager.getLibrarySourceLoader().registerProvider(new FhirLibrarySourceProvider());
303    libraryManager.getLibrarySourceLoader().registerProvider(new DefaultLibrarySourceProvider(Paths.get(folder)));
304
305    loadNamespaces(libraryManager);
306
307    // foreach *.cql file
308    for (File file : new File(folder).listFiles(getCqlFilenameFilter())) {
309      translateFile(modelManager, libraryManager, file, options);
310    }
311  }

313  private void loadNamespaces(LibraryManager libraryManager) {
314    if (namespaceInfo != null) {
315      libraryManager.getNamespaceManager().addNamespace(namespaceInfo);
316    }
317
318    for (NpmPackage p : packages) {
319      if (p.name() != null && !p.name().isEmpty() && p.canonical() != null && !p.canonical().isEmpty()) {
320        NamespaceInfo ni = new NamespaceInfo(p.name(), p.canonical());
321        libraryManager.getNamespaceManager().addNamespace(ni);
322      }
323    }
324  }

344  private ValidationMessage exceptionToValidationMessage(File file, CqlTranslatorException exception) {
345    TrackBack tb = exception.getLocator();
346    if (tb != null) {
347      return new ValidationMessage(ValidationMessage.Source.Publisher, severityToIssueType(exception.getSeverity()),
348              tb.getStartLine(), tb.getStartChar(), tb.getLibrary().getId(), exception.getMessage(),
349              severityToIssueSeverity(exception.getSeverity()));
350    }
351    else {
352      return new ValidationMessage(ValidationMessage.Source.Publisher, severityToIssueType(exception.getSeverity()),
353              file.toString(), exception.getMessage(), severityToIssueSeverity(exception.getSeverity()));
354    }
355  }

357  private void translateFile(ModelManager modelManager, LibraryManager libraryManager, File file, CqlTranslatorOptions options) {
358    logger.logMessage(String.format("Translating CQL source in file %s", file.toString()));
359    CqlSourceFileInformation result = new CqlSourceFileInformation();
360    fileMap.put(file.getAbsoluteFile().toString(), result);
361
362    try {
363
364      // translate toXML
365      CqlTranslator translator = CqlTranslator.fromFile(namespaceInfo, file, modelManager, libraryManager,
366              options.getValidateUnits() ? ucumService : null, options);
367
368      // record errors and warnings
369      for (CqlTranslatorException exception : translator.getExceptions()) {
370        result.getErrors().add(exceptionToValidationMessage(file, exception));
371      }
372
373      if (translator.getErrors().size() > 0) {
374        result.getErrors().add(new ValidationMessage(ValidationMessage.Source.Publisher, IssueType.EXCEPTION, file.getName(),
375                String.format("CQL Processing failed with (%d) errors.", translator.getErrors().size()), IssueSeverity.ERROR));
376        logger.logMessage(String.format("Translation failed with (%d) errors; see the error log for more information.", translator.getErrors().size()));
377      }
378      else {
379        // convert to base64 bytes
380        // NOTE: Publication tooling requires XML content
381        result.setElm(translator.toXml().getBytes());
382        if (options.getFormats().contains(CqlTranslator.Format.JSON)) {
383          result.setJsonElm(translator.toJson().getBytes());
384        }
385        if (options.getFormats().contains(CqlTranslator.Format.JXSON)) {
386          result.setJsonElm(translator.toJxson().getBytes());
387        }
388
389        // TODO: Report context, requires 1.5 translator (ContextDef)
390        // NOTE: In STU3, only Patient context is supported
391
392        // Extract relatedArtifact data (models, libraries, code systems, and value sets)
393        result.relatedArtifacts.addAll(extractRelatedArtifacts(translator.toELM()));
394
395        // Extract parameter data and validate result types are supported types
396        List<ValidationMessage> paramMessages = new ArrayList<>();
397        result.parameters.addAll(extractParameters(translator.toELM(), paramMessages));
398        for (ValidationMessage paramMessage : paramMessages) {
399          result.getErrors().add(new ValidationMessage(paramMessage.getSource(), paramMessage.getType(), file.getName(),
400                  paramMessage.getMessage(), paramMessage.getLevel()));
401        }
402
403        // Extract dataRequirement data
404        result.dataRequirements.addAll(extractDataRequirements(translator.toRetrieves(), translator.getTranslatedLibrary(), libraryManager));
405
406        logger.logMessage("CQL translation completed successfully.");
407      }
408    }
409    catch (Exception e) {
410      result.getErrors().add(new ValidationMessage(ValidationMessage.Source.Publisher, IssueType.EXCEPTION, file.getName(), "CQL Processing failed with exception: "+e.getMessage(), IssueSeverity.ERROR));
411    }
412  }

423  private List<RelatedArtifact> extractRelatedArtifacts(org.hl7.elm.r1.Library library) {
424    List<RelatedArtifact> result = new ArrayList<>();
425
426    // Report model dependencies
427    // URL for a model info is: [baseCanonical]/Library/[model-name]-ModelInfo
428    if (library.getUsings() != null && !library.getUsings().getDef().isEmpty()) {
429      for (UsingDef def : library.getUsings().getDef()) {
430        // System model info is an implicit dependency, do not report
431        if (!def.getLocalIdentifier().equals("System")) {
432          result.add(toRelatedArtifact(def));
433        }
434      }
435    }
436
437    // Report library dependencies
438    if (library.getIncludes() != null && !library.getIncludes().getDef().isEmpty()) {
439      for (IncludeDef def : library.getIncludes().getDef()) {
440        result.add(toRelatedArtifact(def));
441      }
442    }
443
444    // Report CodeSystem dependencies
445    if (library.getCodeSystems() != null && !library.getCodeSystems().getDef().isEmpty()) {
446      for (CodeSystemDef def : library.getCodeSystems().getDef()) {
447        result.add(toRelatedArtifact(def));
448      }
449    }
450
451    // Report ValueSet dependencies
452    if (library.getValueSets() != null && !library.getValueSets().getDef().isEmpty()) {
453      for (ValueSetDef def : library.getValueSets().getDef()) {
454        result.add(toRelatedArtifact(def));
455      }
456    }
457
458    return result;
459  }

461  private List<ParameterDefinition> extractParameters(org.hl7.elm.r1.Library library, List<ValidationMessage> messages) {
462    List<ParameterDefinition> result = new ArrayList<>();
463
464    if (library.getParameters() != null && !library.getParameters().getDef().isEmpty()) {
465      for (ParameterDef def : library.getParameters().getDef()) {
466        result.add(toParameterDefinition(def, messages));
467      }
468    }
469
470    if (library.getStatements() != null && !library.getStatements().getDef().isEmpty()) {
471      for (ExpressionDef def : library.getStatements().getDef()) {
472        if (!(def instanceof FunctionDef) && (def.getAccessLevel() == null || def.getAccessLevel() == AccessModifier.PUBLIC)) {
473          result.add(toOutputParameterDefinition(def, messages));
474        }
475      }
476    }
477
478    return result;
479  }

511  private String getReferenceUrl(String path, String version) {
512    String uri = NamespaceManager.getUriPart(path);
513    String name = NamespaceManager.getNamePart(path);
514
515    if (uri != null) {
516      // The translator has no way to correctly infer the namespace of the FHIRHelpers library, since it will happily provide that source to any namespace that wants it
517      // So override the declaration here so that it points back to the FHIRHelpers library in the base specification
518      if (name.equals("FHIRHelpers") && !uri.equals("http://hl7.org/fhir")) {
519        uri = "http://hl7.org/fhir";
520      }
521      return String.format("%s/Library/%s%s", uri, name, version != null ? ("|" + version) : "");
522    }
523
524    return String.format("Library/%s%s", path, version != null ? ("|" + version) : "");
525  }

539  private ParameterDefinition toParameterDefinition(ParameterDef def, List<ValidationMessage> messages) {
540    org.hl7.cql.model.DataType parameterType = def.getResultType() instanceof ListType ? ((ListType)def.getResultType()).getElementType() : def.getResultType();
541
542    AtomicBoolean isList = new AtomicBoolean(false);
543    Enumerations.FHIRAllTypes typeCode = Enumerations.FHIRAllTypes.fromCode(toFHIRParameterTypeCode(parameterType, def.getName(), isList, messages));
544
545    return new ParameterDefinition()
546            .setName(def.getName())
547            .setUse(Enumerations.OperationParameterUse.IN)
548            .setMin(0)
549            .setMax(isList.get() ? "*" : "1")
550            .setType(typeCode);
551  }

553  private ParameterDefinition toOutputParameterDefinition(ExpressionDef def, List<ValidationMessage> messages) {
554    AtomicBoolean isList = new AtomicBoolean(false);
555    Enumerations.FHIRAllTypes typeCode = Enumerations.FHIRAllTypes.fromCode(toFHIRResultTypeCode(def.getResultType(), def.getName(), isList, messages));
556
557    return new ParameterDefinition()
558            .setName(def.getName())
559            .setUse(Enumerations.OperationParameterUse.OUT)
560            .setMin(0)
561            .setMax(isList.get() ? "*" : "1")
562            .setType(typeCode);
563  }

565  private String toFHIRResultTypeCode(org.hl7.cql.model.DataType dataType, String defName, AtomicBoolean isList, List<ValidationMessage> messages) {
566    AtomicBoolean isValid = new AtomicBoolean(true);
567    String resultCode = toFHIRTypeCode(dataType, isValid, isList);
568    if (!isValid.get()) {
569      // Issue a warning that the result type is not supported
570      messages.add(new ValidationMessage(ValidationMessage.Source.Publisher, IssueType.NOTSUPPORTED, "CQL Library Packaging",
571              String.format("Result type %s of definition %s is not supported; implementations may not be able to use the result of this expression",
572                      dataType.toLabel(), defName), IssueSeverity.WARNING));
573    }
574
575    return resultCode;
576  }

578  private String toFHIRParameterTypeCode(org.hl7.cql.model.DataType dataType, String parameterName, AtomicBoolean isList, List<ValidationMessage> messages) {
579    AtomicBoolean isValid = new AtomicBoolean(true);
580    String resultCode = toFHIRTypeCode(dataType, isValid, isList);
581    if (!isValid.get()) {
582      // Issue a warning that the parameter type is not supported
583      messages.add(new ValidationMessage(ValidationMessage.Source.Publisher, IssueType.NOTSUPPORTED, "CQL Library Packaging",
584              String.format("Parameter type %s of parameter %s is not supported; reported as FHIR.Any", dataType.toLabel(), parameterName), IssueSeverity.WARNING));
585    }
586
587    return resultCode;
588  }

590  private String toFHIRTypeCode(org.hl7.cql.model.DataType dataType, AtomicBoolean isValid, AtomicBoolean isList) {
591    isList.set(false);
592    if (dataType instanceof ListType) {
593      isList.set(true);
594      return toFHIRTypeCode(((ListType)dataType).getElementType(), isValid);
595    }
596
597    return toFHIRTypeCode(dataType, isValid);
598  }

600  private String toFHIRTypeCode(org.hl7.cql.model.DataType dataType, AtomicBoolean isValid) {
601    isValid.set(true);
602    if (dataType instanceof NamedType) {
603      switch (((NamedType)dataType).getName()) {
604        case "System.Boolean": return "boolean";
605        case "System.Integer": return "integer";
606        case "System.Decimal": return "decimal";
607        case "System.Date": return "date";
608        case "System.DateTime": return "dateTime";
609        case "System.Time": return "time";
610        case "System.String": return "string";
611        case "System.Quantity": return "Quantity";
612        case "System.Ratio": return "Ratio";
613        case "System.Any": return "Any";
614        case "System.Code": return "Coding";
615        case "System.Concept": return "CodeableConcept";
616      }
617
618      if ("FHIR".equals(((NamedType)dataType).getNamespace())) {
619        return ((NamedType)dataType).getSimpleName();
620      }
621    }
622
623    if (dataType instanceof IntervalType) {
624      if (((IntervalType)dataType).getPointType() instanceof NamedType) {
625        switch (((NamedType)((IntervalType)dataType).getPointType()).getName()) {
626          case "System.Date":
627          case "System.DateTime": return "Period";
628          case "System.Quantity": return "Range";
629        }
630      }
631    }
632
633    isValid.set(false);
634    return "Any";
635  }

637  private org.hl7.fhir.r5.model.DataRequirement toDataRequirement(Retrieve retrieve, TranslatedLibrary library, LibraryManager libraryManager) {
638    org.hl7.fhir.r5.model.DataRequirement dr = new org.hl7.fhir.r5.model.DataRequirement();
639
640    dr.setType(org.hl7.fhir.r5.model.Enumerations.FHIRAllTypes.fromCode(retrieve.getDataType().getLocalPart()));
641
642    // Set profile if specified
643    if (retrieve.getTemplateId() != null) {
644      dr.setProfile(Collections.singletonList(new org.hl7.fhir.r5.model.CanonicalType(retrieve.getTemplateId())));
645    }
646
647    // Set code path if specified
648    if (retrieve.getCodeProperty() != null) {
649      org.hl7.fhir.r5.model.DataRequirement.DataRequirementCodeFilterComponent cfc =
650              new org.hl7.fhir.r5.model.DataRequirement.DataRequirementCodeFilterComponent();
651
652      cfc.setPath(retrieve.getCodeProperty());
653
654      // TODO: Support retrieval when the target is a CodeSystemRef
655
656      if (retrieve.getCodes() instanceof ValueSetRef) {
657        ValueSetRef vsr = (ValueSetRef)retrieve.getCodes();
658        cfc.setValueSet(toReference(resolveValueSetRef(vsr, library, libraryManager)));
659      }
660
661      if (retrieve.getCodes() instanceof org.hl7.elm.r1.ToList) {
662        org.hl7.elm.r1.ToList toList = (org.hl7.elm.r1.ToList)retrieve.getCodes();
663        resolveCodeFilterCodes(cfc, toList.getOperand(), library, libraryManager);
664      }
665
666      if (retrieve.getCodes() instanceof org.hl7.elm.r1.List) {
667        org.hl7.elm.r1.List codeList = (org.hl7.elm.r1.List)retrieve.getCodes();
668        for (Expression e : codeList.getElement()) {
669          resolveCodeFilterCodes(cfc, e, library, libraryManager);
670        }
671      }
672
673      dr.getCodeFilter().add(cfc);
674    }
675
676    // TODO: Set date range filters if literal
677
678    return dr;
679  }

681  private void resolveCodeFilterCodes(org.hl7.fhir.r5.model.DataRequirement.DataRequirementCodeFilterComponent cfc, Expression e,
682                                      TranslatedLibrary library, LibraryManager libraryManager) {
683    if (e instanceof org.hl7.elm.r1.CodeRef) {
684      CodeRef cr = (CodeRef)e;
685      cfc.addCode(toCoding(toCode(resolveCodeRef(cr, library, libraryManager)), library, libraryManager));
686    }
687
688    if (e instanceof org.hl7.elm.r1.Code) {
689      cfc.addCode(toCoding((org.hl7.elm.r1.Code)e, library, libraryManager));
690    }
691
692    if (e instanceof org.hl7.elm.r1.ConceptRef) {
693      ConceptRef cr = (ConceptRef)e;
694      org.hl7.fhir.r5.model.CodeableConcept c = toCodeableConcept(toConcept(resolveConceptRef(cr, library, libraryManager), library, libraryManager), library, libraryManager);
695      for (org.hl7.fhir.r5.model.Coding code : c.getCoding()) {
696        cfc.addCode(code);
697      }
698    }
699
700    if (e instanceof org.hl7.elm.r1.Concept) {
701      org.hl7.fhir.r5.model.CodeableConcept c = toCodeableConcept((org.hl7.elm.r1.Concept)e, library, libraryManager);
702      for (org.hl7.fhir.r5.model.Coding code : c.getCoding()) {
703        cfc.addCode(code);
704      }
705    }
706  }

708  private org.hl7.fhir.r5.model.Coding toCoding(Code code, TranslatedLibrary library, LibraryManager libraryManager) {
709    CodeSystemDef codeSystemDef = resolveCodeSystemRef(code.getSystem(), library, libraryManager);

711    coding.setCode(code.getCode());
712    coding.setDisplay(code.getDisplay());
713    coding.setSystem(codeSystemDef.getId());
714    coding.setVersion(codeSystemDef.getVersion());
715    return coding;

718  private org.hl7.fhir.r5.model.CodeableConcept toCodeableConcept(Concept concept, TranslatedLibrary library, LibraryManager libraryManager) {
719    org.hl7.fhir.r5.model.CodeableConcept codeableConcept = new org.hl7.fhir.r5.model.CodeableConcept();

790  private TranslatedLibrary resolveLibrary(LibraryManager libraryManager, VersionedIdentifier libraryIdentifier) {
791    if (libraryManager.getTranslatedLibraries().containsKey(libraryIdentifier.getId())) {
792      return libraryManager.getTranslatedLibraries().get(libraryIdentifier.getId());
793    }
794
795    throw new IllegalArgumentException(String.format("Could not resolve reference to translated library %s", libraryIdentifier.getId()));
796  }
