090  private void loadPaths(JsonObject igs) throws Exception {
091    JsonElement e = igs.get("path-pattern");
092    if (e != null)
093      pathPattern = e.getAsString(); 
094    defaultConfig = igs.getAsJsonObject("defaults");
095    resourceConfig = igs.getAsJsonObject("resources");
096    if (resourceConfig != null) {
097      for (Entry<String, JsonElement> pp : resourceConfig.entrySet()) {
098        if (pp.getKey().equals("*")) {
099          autoPath = true;
100        } else if (!pp.getKey().startsWith("_")) {
101          String s = pp.getKey();
102          if (!s.contains("/"))
103            throw new Exception("Bad Resource Identity - should have the format [Type]/[id]:" + s);
104          String type = s.substring(0,  s.indexOf("/"));
105          String id = s.substring(s.indexOf("/")+1); 
106          if (!context.hasResource(StructureDefinition.class, "http://hl7.org/fhir/StructureDefinition/"+type) && !(context.hasResource(StructureDefinition.class , "http://hl7.org/fhir/StructureDefinition/Conformance") && type.equals("CapabilityStatement")))
107            throw new Exception("Bad Resource Identity - should have the format [Type]/[id] where Type is a valid resource type:" + s);
108          if (!id.matches(FormatUtilities.ID_REGEX))
109            throw new Exception("Bad Resource Identity - should have the format [Type]/[id] where id is a valid FHIR id type:" + s);
110
111          if (!(pp.getValue() instanceof JsonObject))
112            throw new Exception("Unexpected type in resource list - must be an object");
113          JsonObject o = (JsonObject) pp.getValue();
114          JsonElement p = o.get("base");
115          //        if (p == null)
116          //          throw new Exception("You must provide a base on each path in the json file");
117          if (p != null && !(p instanceof JsonPrimitive) && !((JsonPrimitive) p).isString())
118            throw new Exception("Unexpected type in paths - base must be a string");
119          p = o.get("defns");
120          if (p != null && !(p instanceof JsonPrimitive) && !((JsonPrimitive) p).isString())
121            throw new Exception("Unexpected type in paths - defns must be a string");
122          p = o.get("source");
123          if (p != null && !(p instanceof JsonPrimitive) && !((JsonPrimitive) p).isString())
124            throw new Exception("Unexpected type in paths - source must be a string");
125        }
126      }
127    } else if (template == null)
128      throw new Exception("No \"resources\" entry found in json file (see http://wiki.hl7.org/index.php?title=IG_Publisher_Documentation#Control_file)");
129  }

154  public String doReplacements(String s, FetchedResource r, Map<String, String> vars, String format) throws FHIRException {
155    if (Utilities.noString(s))
156      return s;
157    if (r.getId()== null) {
158      throw new FHIRException("Error doing replacements - no id defined in resource: " + (r.getTitle()== null ? "NO TITLE EITHER" : r.getTitle()));
159    }
160    s = s.replace("{{[title]}}", r.getTitle() == null ? "?title?" : r.getTitle());
161    s = s.replace("{{[name]}}", r.getId()+(format==null? "": "-"+format)+"-html");
162    s = s.replace("{{[id]}}", r.getId());
163    if (format!=null)
164      s = s.replace("{{[fmt]}}", format);
165    s = s.replace("{{[type]}}", r.fhirType());
166    s = s.replace("{{[uid]}}", r.fhirType()+"="+r.getId());
167    if (vars != null) {
168      for (String n : vars.keySet()) {
169        String v = vars.get(n);
170        if (v == null) {
171          v = "";
172        }
173        s = s == null ? "" : s.replace("{{["+n+"]}}", v);
174      }
175    }
176    return s;
177  }

221  public boolean wantGen(FetchedResource r, String code) {
222    if (r.getConfig() != null && hasBoolean(r.getConfig(), code))
223      return getBoolean(r.getConfig(), code);
224    JsonObject cfg = null;
225    if (defaultConfig != null) {
226      cfg = defaultConfig.getAsJsonObject(r.fhirType());
227      if (cfg != null && hasBoolean(cfg, code)) {
228        return getBoolean(cfg, code);
229      }
230      cfg = defaultConfig.getAsJsonObject("Any");
231      if (cfg != null && hasBoolean(cfg, code)) {
232        return getBoolean(cfg, code);
233      }
234    }
235    return true;
236  }

344  public void loadSpecPaths(SpecMapManager paths) throws Exception {
345    this.specPaths = paths;
346    for (CanonicalResource bc : context.allConformanceResources()) {
347      String s = getOverride(bc.getUrl());
348      if (s == null) {
349        s = paths.getPath(bc.getUrl(), bc.getMeta().getSource());
350      }
351      if (s == null && bc instanceof CodeSystem) { // work around for an R2 issue) 
352        CodeSystem cs = (CodeSystem) bc;
353        s = paths.getPath(cs.getValueSet(), null);
354      }
355      if (s != null) {
356        bc.setUserData("path", specPath(s));
357      // special cases
358      } else if (bc.hasUrl() && bc.getUrl().equals("http://hl7.org/fhir/ValueSet/security-role-type")) {
359        bc.setUserData("path", specPath("valueset-security-role-type.html"));
360      } else if (bc.hasUrl() && bc.getUrl().equals("http://hl7.org/fhir/ValueSet/object-lifecycle-events")) {
361        bc.setUserData("path", specPath("valueset-object-lifecycle-events.html"));
362      } else if (bc.hasUrl() && bc.getUrl().equals("http://hl7.org/fhir/ValueSet/performer-function")) {
363        bc.setUserData("path", specPath("valueset-performer-function.html"));
364      } else if (bc.hasUrl() && bc.getUrl().equals("http://hl7.org/fhir/ValueSet/written-language")) {
365        bc.setUserData("path", specPath("valueset-written-language.html"));
366        
367//      else
368//        System.out.println("No path for "+bc.getUrl());
369      }
370    }    
371  }

373  private String getOverride(String url) {
374    if ("http://hl7.org/fhir/StructureDefinition/Reference".equals(url))
375      return "references.html#Reference";
376    if ("http://hl7.org/fhir/StructureDefinition/DataRequirement".equals(url))
377     return "metadatatypes.html#DataRequirement";
378    if ("http://hl7.org/fhir/StructureDefinition/ContactDetail".equals(url))
379      return "metadatatypes.html#ContactDetail";
380    if ("http://hl7.org/fhir/StructureDefinition/Contributor".equals(url))
381      return "metadatatypes.html#Contributor";
382    if ("http://hl7.org/fhir/StructureDefinition/ParameterDefinition".equals(url))
383      return "metadatatypes.html#ParameterDefinition";
384    if ("http://hl7.org/fhir/StructureDefinition/RelatedArtifact".equals(url))
385      return "metadatatypes.html#RelatedArtifact";
386    if ("http://hl7.org/fhir/StructureDefinition/TriggerDefinition".equals(url))
387      return "metadatatypes.html#TriggerDefinition";
388    if ("http://hl7.org/fhir/StructureDefinition/UsageContext".equals(url))
389      return "metadatatypes.html#UsageContext";
390    if ("http://hl7.org/fhir/StructureDefinition/Extension".equals(url))
391      return "extensibility.html#Extension";
392    return null;
393  }

407  public void findConfiguration(FetchedFile f, FetchedResource r) {
408    if (template != null) {
409      JsonObject cfg = null;
410      if (r.isExample()) {
411        cfg = defaultConfig.getAsJsonObject("example");
412      }        
413      if (cfg == null && r.fhirType().equals("StructureDefinition")) {
414        cfg = defaultConfig.getAsJsonObject(r.fhirType()+":"+getSDType(r));
415      }
416      if (cfg == null)
417        cfg = template.getConfig(r.fhirType(), r.getId());        
418      r.setConfig(cfg);
419    }
420    if (r.getConfig() == null && resourceConfig != null) {
421      JsonObject e = resourceConfig.getAsJsonObject(r.fhirType()+"/"+r.getId());
422      if (e != null)
423        r.setConfig(e);
424    }
425  }

427  public void checkForPath(FetchedFile f, FetchedResource r, CanonicalResource bc, boolean inner) throws FHIRException {
428    if (!bc.hasUrl())
429      error(f, bc.fhirType()+".url", "Resource has no url: "+bc.getId(), I18nConstants.RESOURCE_ID_NO_URL);
430    else if (bc.getUrl().startsWith(canonical) && !bc.getUrl().endsWith("/"+bc.getId()) && !listedURLExemptions.contains(bc.getUrl()))
431      error(f, bc.fhirType()+".url","Resource id/url mismatch: "+bc.getId()+"/"+bc.getUrl(), I18nConstants.RESOURCE_ID_MISMATCH);
432    if (!inner && !r.getId().equals(bc.getId()))
433      error(f, bc.fhirType()+".id", "Resource id/loaded id mismatch: "+r.getId()+"/"+bc.getUrl(), I18nConstants.RESOURCE_ID_LOADED_MISMATCH);
434    if (r.getConfig() == null)
435      findConfiguration(f, r);
436    JsonObject e = r.getConfig();
437    bc.setUserData("config", e);
438    String base = getProperty(r,  "base");
439    if (base != null) 
440      bc.setUserData("path", doReplacements(base, r, null, null));
441    else if (pathPattern != null)
442      bc.setUserData("path", pathPattern.replace("[type]", r.fhirType()).replace("[id]", r.getId()));
443    else
444      bc.setUserData("path", r.fhirType()+"/"+r.getId()+".html");
445    for (Resource cont : bc.getContained()) {
446      if (base != null) 
447        cont.setUserData("path", doReplacements(base, r, cont, null, null, bc.getId()+"_"));
448      else if (pathPattern != null)
449        cont.setUserData("path", pathPattern.replace("[type]", r.fhirType()).replace("[id]", bc.getId()+"_"+cont.getId()));
450      else
451        cont.setUserData("path", r.fhirType()+"/"+bc.getId()+"_"+r.getId()+".html");
452    }
453  }

502    StructureDefinition sd = context.fetchTypeDefinition(name);
503    return sd != null && (sd.getKind() == StructureDefinitionKind.PRIMITIVETYPE || sd.getKind() == StructureDefinitionKind.COMPLEXTYPE) && sd.getDerivation() == TypeDerivationRule.SPECIALIZATION;

510  }

545  public BindingResolution resolveBinding(StructureDefinition profile, String ref, String path) {
546    BindingResolution br = new BindingResolution();
547    if (ref.startsWith("http://hl7.org/fhir/ValueSet/v3-")) {
548      br.url = specPath("v3/"+ref.substring(32)+"/vs.html");
549      br.display = ref.substring(32);
550    } else if (ref.startsWith("ValueSet/")) {
551      ValueSet vs = context.fetchResource(ValueSet.class, makeCanonical(ref));
552      if (vs == null) {
553        br.url = ref;  
554        if (ref.equals("http://tools.ietf.org/html/bcp47"))
555          br.display = "IETF BCP-47";
556        else if (ref.equals("http://www.rfc-editor.org/bcp/bcp13.txt"))
557          br.display = "IETF BCP-13";
558        else if (ref.equals("http://www.ncbi.nlm.nih.gov/nuccore?db=nuccore"))
559          br.display = "NucCore";
560        else if (ref.equals("https://rtmms.nist.gov/rtmms/index.htm#!rosetta"))
561          br.display = "Rosetta";
562        else if (ref.equals("http://www.iso.org/iso/country_codes.htm"))
563          br.display = "ISO Country Codes";
564        else {
565          br.url = ref.substring(9)+".html"; // broken link, 
566          br.display = ref.substring(9);
567          brokenLinkWarning(path, ref);
568        }
569      } else {
570        br.url = vs.getUserString("path");
571        br.display = vs.getName(); 
572      }
573    } else { 
574      if (ref.startsWith("http://hl7.org/fhir/ValueSet/")) {
575        ValueSet vs = context.fetchResource(ValueSet.class, ref);
576        if (vs != null) { 
577          br.url = vs.getUserString("path");
578          br.display = vs.getName(); 
579        } else {
580          String vsr = VersionConvertorConstants.vsToRef(ref);
581          if (vsr != null) {
582            br.display = ref.substring(29);
583            br.url = vsr;
584          } else {
585            br.display = ref.substring(29);
586            br.url = ref.substring(29)+".html";
587            brokenLinkWarning(path, ref);
588          }
589        }
590      } else if (ref.startsWith("http://hl7.org/fhir/ValueSet/v3-")) {
591        br.url = specPath("v3/"+ref.substring(26)+"/index.html"); 
592        br.display = ref.substring(26);
593      } else if (ref.startsWith("http://hl7.org/fhir/ValueSet/v2-")) {
594        br.url = specPath("v2/"+ref.substring(26)+"/index.html"); 
595        br.display = ref.substring(26);
596      } else if (ref.startsWith("http://loinc.org/vs/")) {
597        String code = tail(ref);
598        if (code.startsWith("LL")) {
599          br.url = "https://r.details.loinc.org/AnswerList/"+code+".html";
600          br.display = "LOINC Answer List "+code;
601        } else {
602          br.url = "https://r.details.loinc.org/LOINC/"+code+".html";
603          br.display = "LOINC "+code;
604        }
605      } else {
606        ValueSet vs = context.fetchResource(ValueSet.class, ref);
607        if (vs != null) {
608          if (ref.contains("|")) {
609            br.url = vs.getUserString("versionpath");
610            if (br.url == null) {
611              br.url = vs.getUserString("path");
612            }
613            br.display = vs.getName() + " (" + vs.getVersion() + ")"; 
614          } else if (vs != null && vs.hasUserData("path")) {
615            br.url = vs.getUserString("path");  
616            br.display = vs.present();
617          } else {
618            br.url = vs.getUserString("path");
619            br.display = vs.getName(); 
620          }
621        } else if (ref.startsWith("http://cts.nlm.nih.gov/fhir/ValueSet/")) {          
622          String oid = ref.substring("http://cts.nlm.nih.gov/fhir/ValueSet/".length());
623          br.url = "https://vsac.nlm.nih.gov/valueset/"+oid+"/expansion";  
624          br.display = "VSAC "+oid;
625        } else if (Utilities.isAbsoluteUrl(ref) && (!ref.startsWith("http://hl7.org") || !ref.startsWith("http://terminology.hl7.org"))) {
626          br.url = Utilities.encodeUri(ref);  
627          br.display = ref;
628        } else if (vs == null) {
629          br.url = ref+".html"; // broken link, 
630          br.display = ref;
631          brokenLinkWarning(path, ref);
632        }
633      }
634    }
635    return br;
636  }

638  private String tail(String ref) {
639    if  (ref.contains("/"))
640      return ref.substring(ref.lastIndexOf("/")+1);
641    else
642      return ref;
643  }

682  public String getLinkFor(FetchedResource r, boolean replace) {
683    String base = getProperty(r, "base");
684    if (base!=null) {
685      if (replace) {
686        base = base.replace("{{[id]}}", r.getId());
687        base = base.replace("{{[type]}}", r.fhirType());
688      }
689      return base;
690    }
691    return r.fhirType()+"-"+r.getId()+".html";
692  }

694  public String getLinkFor(FetchedResource r, boolean replace, Resource contained) {
695    String base = getProperty(r, "base");
696    if (base!=null) {
697      if (replace) {
698        base = base.replace("{{[id]}}", r.getId());
699        base = base.replace("{{[type]}}", r.fhirType());
700      }
701      base = base.replace(".html", "_"+contained.getId()+".html");
702      return base;
703    }
704    return r.fhirType()+"-"+r.getId()+"_"+contained.getId()+".html";
705  }

715  public BindingResolution resolveActualUrl(String uri) {
716    BindingResolution br = new BindingResolution();
717    if (uri != null) {
718      if (uri.startsWith("http://loinc.org/vs/")) {
719        String code = tail(uri);
720        if (code.startsWith("LL")) {
721          br.url = "https://r.details.loinc.org/AnswerList/"+code+".html";
722          br.display = "LOINC Answer List "+code;
723        } else {
724          br.url = "https://r.details.loinc.org/LOINC/"+code+".html";
725          br.display = "LOINC "+code;
726        }
727      } else if (uri.startsWith("urn:")) {
728        br.url = null;
729        br.display = uri;
730      } else {
731        br.url = uri;
732        br.display = uri;
733      }
734    }
735    return br;
736  }
