087    private String getSpecName(String url) {
088      if (url.startsWith("http://hl7.org/fhir/StructureDefinition/")) {
089        return "<a href=\"http://hl7.org/fhir\">fhir</a>.";
090      }
091      PackageVersion pck = context.getPackageForUrl(url);
092      if (pck != null) {
093        if (pck.getId().equals(packageId)) {
094          return "<i>this.</i>";
095        }
096        PackageDetails det = context.getPackage(pck);
097        if (det != null) {
098          return "<a href=\""+det.getWeb()+"\">"+det.getName()+"</a>.";
099        }
100        return pck.getId()+".";
101      }
102      return "";
103    }

116      if (e.hasMessages()) {
117        b.append("<li>"+root+e.getPath()+": Validated against "+analyse(e.getMessages())+"</li>");
118      }
119      for (Element c : e.getChildren()) {
120        showMessages(b, root, c);
121      }

132    public String analyse(List<ValidationMessage> vmlist) {
133      String base = context.formatMessage(I18nConstants.VALIDATION_VAL_PROFILE_SIGNPOST_BASE);
134      List<String> list = new ArrayList<>();
135      List<String> others = new ArrayList<>();
136      for (ValidationMessage vm : vmlist) {
137        if (vm.getMessage().startsWith(base)) {
138          String s = vm.getMessage().substring(base.length()).trim();
139          String url = s.contains(" ") ? s.substring(0, s.indexOf(" ")) : s;
140          s = s.contains(" ") ? s.substring(s.indexOf(" ")) : "";
141          String specName = getSpecName(url);
142          StructureDefinition sd = context.fetchResource(StructureDefinition.class, url);
143          String l = null;
144          if (sd != null) {
145            l = specName+"<a href=\""+sd.getUserString("path")+"\">"+sd.present()+"</a>"+s;
146          } else {
147            l = "<a href=\""+url+"\">"+url+"</a>"+s;          
148          } 
149          if (!list.contains(l)) {
150            list.add(l);
151            count++;
152          }
153        } else {
154          others.add(vm.getMessage());
155        }
156      }
157      return list(list);
158    }

289  private void determineCode() {
290    if (provider.getCanonical().startsWith("http://hl7.org/fhir")) {
291      String[] u = provider.getCanonical().split("\\/");
292      String ucode = u[u.length-1];
293      String[] p = packageId.split("\\.");
294      String pcode = p[p.length-1];
295      igcode = ucode;
296      if (!ucode.equals(pcode)) {
297        igCodeError = "Error: codes in canonical and package id are different: "+ucode+" vs "+pcode+".";
298      }
299      igrealm = p[p.length-2];
300      String urealm = u[u.length-2];
301      if (!igrealm.equals(urealm)) {
302        igRealmError = "Error: realms in canonical and package id are different: "+igrealm+" vs "+urealm;
303      } else if (!igrealm.equalsIgnoreCase(realm.code())) {
304        igRealmError = "Error: realms in IG definition and package id are different: "+igrealm+" vs "+realm.code();
305      }
306    } else {
307      this.igcode = "n/a";
308    }
309  }

327  public String generate(String title, List<ValidationMessage> allErrors, List<FetchedFile> files, String path, SuppressedMessageInformation filteredMessages) throws IOException {
328    
329    for (FetchedFile f : files) {
330      for (ValidationMessage vm : filterMessages(f.getErrors(), false, filteredMessages)) {
331        if (vm.getLevel().equals(ValidationMessage.IssueSeverity.FATAL)||vm.getLevel().equals(ValidationMessage.IssueSeverity.ERROR))
332          err++;
333        else if (vm.getLevel().equals(ValidationMessage.IssueSeverity.WARNING))
334          warn++;
335        else if (!vm.isSignpost()) {
336          info++;
337        }
338      }
339    }
340    
341    List<ValidationMessage> linkErrors = filterMessages(allErrors, true, filteredMessages); 
342    for (ValidationMessage vm : linkErrors) {
343      if (vm.getSource() == Source.LinkChecker) {
344        link++;
345      } else if (vm.getLevel().equals(ValidationMessage.IssueSeverity.FATAL)||vm.getLevel().equals(ValidationMessage.IssueSeverity.ERROR))
346        err++;
347      else if (vm.getLevel().equals(ValidationMessage.IssueSeverity.WARNING))
348        warn++;
349      else if (!vm.isSignpost()) {
350        info++;
351      }
352    }
353    
354    files = genQAHtml(title, files, path, filteredMessages, linkErrors, true);
355    files = genQAHtml(title, files, path, filteredMessages, linkErrors, false);
356
357    Bundle validationBundle = new Bundle().setType(Bundle.BundleType.COLLECTION);
358    OperationOutcome oo = new OperationOutcome();
359    validationBundle.addEntry(new BundleEntryComponent().setResource(oo));
360    for (ValidationMessage vm : linkErrors) {
361      if (vm.getSource() != Source.LinkChecker && vm.getLocation()!=null) {
362        FHIRPathEngine fpe = new FHIRPathEngine(provider.getContext());
363        try {
364          fpe.parse(vm.getLocation());
365        } catch (Exception e) {
366          System.out.println("Internal error in location for message: '"+e.getMessage()+"', loc = '"+subst100(vm.getLocation())+"', err = '"+subst100(vm.getMessage())+"'");
367        }
368        oo.getIssue().add(OperationOutcomeUtilities.convertToIssue(vm, oo));
369      }
370    }
371    for (FetchedFile f : files) {
372      if (!f.getErrors().isEmpty()) {
373        oo = new OperationOutcome();
374        validationBundle.addEntry(new BundleEntryComponent().setResource(oo));
375        ToolingExtensions.addStringExtension(oo, ToolingExtensions.EXT_OO_FILE, f.getName());
376        for (ValidationMessage vm : filterMessages(f.getErrors(), false, filteredMessages)) {
377          oo.getIssue().add(OperationOutcomeUtilities.convertToIssue(vm, oo));
378        }
379      }
380    }
381    FileOutputStream s = new FileOutputStream(Utilities.changeFileExt(path, ".xml"));
382    new XmlParser().compose(s, validationBundle, true);
383    s.close();
384
385    genQAText(title, files, path, filteredMessages, linkErrors);
386    
387    String summary = "Errors: " + err + ", Warnings: " + warn + ", Info: " + info+", Broken Links = "+link;
388    return path + "\r\n" + summary;
389  }

391  public void genQAText(String title, List<FetchedFile> files, String path, SuppressedMessageInformation filteredMessages, List<ValidationMessage> linkErrors)
392      throws IOException {
393    StringBuilder b = new StringBuilder();
394    b.append(genHeaderTxt(title, err, warn, info));
395    b.append(genSummaryRowTxtInternal(linkErrors));
396    files = sorted(files);
397    for (FetchedFile f : files) 
398      b.append(genSummaryRowTxt(f));
399    b.append(genEnd());
400    b.append(genStartTxtInternal());
401    for (ValidationMessage vm : linkErrors)
402      b.append(vm.getDisplay() + "\r\n");
403    b.append(genEndTxt());
404    for (FetchedFile f : files) {
405      b.append(genStartTxt(f));
406      for (ValidationMessage vm : filterMessages(f.getErrors(), false, filteredMessages))
407        b.append(vm.getDisplay() + "\r\n");
408      b.append(genEndTxt());
409    }    
410    b.append(genFooterTxt(title));
411    TextFile.stringToFile(b.toString(), Utilities.changeFileExt(path, ".txt"));
412  }

414  public List<FetchedFile> genQAHtml(String title, List<FetchedFile> files, String path, SuppressedMessageInformation filteredMessages, List<ValidationMessage> linkErrors, boolean allIssues) throws IOException {
415    StringBuilder b = new StringBuilder();
416    b.append(genHeader(title, err, warn, info, link, filteredMessages.count(), allIssues, path));
417    b.append(genSummaryRowInteral(linkErrors));
418    files = sorted(files);
419    for (FetchedFile f : files) {
420      if (allIssues || hasIssues(f, filteredMessages)) {
421        b.append(genSummaryRow(f, filteredMessages));
422      }
423    }
424    b.append(genEnd());
425    b.append(genStartInternal());
426    int id = 0;
427    for (ValidationMessage vm : linkErrors) {
428      b.append(genDetails(vm, id));
429      id++;
430    }
431    b.append(genEnd());
432    int i = 0;
433    for (FetchedFile f : files) {
434      if (allIssues || hasIssues(f, filteredMessages)) {
435        i++;
436        b.append(genStart(f, i));
437        if (countNonSignpostMessages(f, filteredMessages))
438          b.append(startTemplateErrors);
439        else
440          b.append(startTemplateNoErrors);
441        for (ValidationMessage vm : filterMessages(f.getErrors(), false, filteredMessages)) {
442          b.append(genDetails(vm, id));
443          id++;
444        }
445        b.append(genEnd());
446      }
447    }    
448    b.append(genSuppressedMessages(filteredMessages));
449    b.append("<a name=\"sorted\"> </a>\r\n<p><b>Errors sorted by type</b></p>\r\n");
450    for (String n : messageIdNames()) {
451      List<FiledValidationMessage> fvml = new ArrayList<>();    
452      for (FetchedFile f : files) {
453        getMatchingMessages(f, n, fvml, filteredMessages);
454      }
455      if (fvml.size() > 0) {
456        b.append(genGroupStart(n));
457        for (FiledValidationMessage fvm : fvml) {
458          b.append(genGroupDetails(fvm.getF(), fvm.getVm()));
459        }
460        b.append(genGroupEnd());
461      }
462    }
463    b.append(genFooter(title));
464    TextFile.stringToFile(b.toString(), allIssues ? path : Utilities.changeFileExt(path, ".min.html"));
465    return files;
466  }

470  private boolean countNonSignpostMessages(FetchedFile f, SuppressedMessageInformation filteredMessages) {
471    List<ValidationMessage> uniqueErrors = filterMessages(f.getErrors(), false, filteredMessages);
472    for (ValidationMessage vm : uniqueErrors) {
473      if (!vm.isSignpost()) {
474        return true;
475      }
476    }
477    return false;
478  }

480  private boolean hasIssues(FetchedFile f, SuppressedMessageInformation filteredMessages) {
481    List<ValidationMessage> uniqueErrors = filterMessages(f.getErrors(), false, filteredMessages);
482    for (ValidationMessage vm : uniqueErrors) {
483      if (vm.getLevel() != IssueSeverity.INFORMATION) {
484        return true;
485      }
486    }
487    return false;
488  }

490  private void getMatchingMessages(FetchedFile f, String n, List<FiledValidationMessage> fvml, SuppressedMessageInformation filteredMessages) {
491    for (ValidationMessage vm : filterMessages(f.getErrors(), false, filteredMessages)) {
492      if (n.equals(vm.getMessageId()) && !vm.isSignpost()) {
493        fvml.add(new FiledValidationMessage(f, vm));
494      }
495    }

498  private List<String> messageIdNames() {
499    I18nConstants obj = new I18nConstants();
500    org.hl7.fhir.igtools.publisher.I18nConstants obj2 = new org.hl7.fhir.igtools.publisher.I18nConstants(); // not that it really matters?
501    List<String> names = new ArrayList<>();
502    Field[] interfaceFields=I18nConstants.class.getFields();
503    for(Field f : interfaceFields) {
504      try {
505        if (Modifier.isStatic(f.getModifiers())) {
506          String n = (String) f.get(obj);
507          names.add(n);
508        }
509      } catch (Exception e) {
510      }
511    }
512    interfaceFields=org.hl7.fhir.igtools.publisher.I18nConstants.class.getFields();
513    for(Field f : interfaceFields) {
514      try {
515        if (Modifier.isStatic(f.getModifiers())) {
516          String n = (String) f.get(obj2);
517          names.add(n);
518        }
519      } catch (Exception e) {
520      }
521    }
522    Collections.sort(names);
523    return names;
524  }

532  private String genSuppressedMessages(SuppressedMessageInformation msgs) {
533    StringBuilder b = new StringBuilder();
534    b.append("<a name=\"suppressed\"> </a>\r\n<p><b>Suppressed Messages (Warnings, hints, broken links)</b></p>\r\n");
535    boolean found = false;
536    for (String s : msgs.categories()) {
537      b.append("<p><b>"+Utilities.escapeXml(s)+"</b></p><ul>\r\n");
538      for (String m : msgs.list(s)) {
539        b.append(" <li>"+Utilities.escapeXml(m)+"</li>\r\n");
540      }
541      b.append("</ul>\r\n");
542    }
543    if (!found) {
544      b.append("<p>No suppressed messsages</p>");
545    }
546    return b.toString();
547  }

550  public static List<ValidationMessage> filterMessages(List<ValidationMessage> messages, boolean canSuppressErrors, SuppressedMessageInformation suppressedMessages) {
551    List<ValidationMessage> passList = new ArrayList<ValidationMessage>();
552    Set<String> msgs = new HashSet<>();
553    for (ValidationMessage message : messages) {
554      boolean passesFilter = true;
555      if (canSuppressErrors || !message.getLevel().isError()) {
556        if (suppressedMessages.contains(message.getDisplay()) || suppressedMessages.contains(message.getMessage()) || suppressedMessages.contains(message.getHtml()) || suppressedMessages.contains(message.getMessageId()) ) {
557          passesFilter = false;
558        } else if (msgs.contains(message.getLocation()+"|"+message.getMessage())) {
559          passesFilter = false;
560        }
561      }
562      if (NO_FILTER) {
563        passesFilter = true;
564      }
565      if (message.isSignpost()) {
566        passesFilter = false;        
567      }
568      if (passesFilter) {
569        if (message.getMessage().contains("australian-states-territories-2")) {
570          System.out.print(".");
571        }
572        passList.add(message);
573        msgs.add(message.getLocation()+"|"+message.getMessage());        
574      } else {
575      }
576    }
577    return passList;
578  }

834  private String checkCopyRightYear() {
835    if (copyrightYear == null) {
836      return "<br/>The IG resource does not contain a copyrightYear parameter.";
837    } else if (copyrightYear.endsWith("+") && copyrightYear.length() == 5) {
838      String s = copyrightYear.substring(0, 4);
839      if (Utilities.isInteger(s)) {
840        int y = Integer.parseInt(s);
841        if (y > Calendar.getInstance().get(Calendar.YEAR)) {
842          return "<br/><br/><span style=\"background-color: #ffcccc\">The copyrightYear parameter ('"+copyrightYear+"') in the IG resource looks wrong - should not be after this year</span>";                  
843        } else {
844          return "<br/><br/>The copyrightYear parameter ('"+copyrightYear+"') in the IG resource is good";          
845        }
846      } else {
847        return "<br/><br/><span style=\"background-color: #ffcccc\">The copyrightYear parameter ('"+copyrightYear+"') in the IG resource can't be understood - expecting YYYY+</span>";        
848      }
849    } else {
850      return "<br/><br/><span style=\"background-color: #ffcccc\">The copyrightYear parameter ('"+copyrightYear+"') in the IG resource can't be understood - expecting YYYY+</span>";        
851    }
852  }
