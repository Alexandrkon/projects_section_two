097    public String summary() {
098        try {
099            if (sd.getDifferential() == null)
100                return "<p>" + translate("sd.summary", "No Summary, as this profile has no differential") + "</p>";
101
102            // references
103            List<String> refs = new ArrayList<String>(); // profile references
104            // extensions (modifier extensions)
105            List<String> ext = new ArrayList<String>(); // extensions
106            // slices
107            List<String> slices = new ArrayList<String>(); // Fixed Values
108            // numbers - must support, required, prohibited, fixed
109            int supports = 0;
110            int requiredOutrights = 0;
111            int requiredNesteds = 0;
112            int fixeds = 0;
113            int prohibits = 0;
114
115            for (ElementDefinition ed : sd.getDifferential().getElement()) {
116                if (ed.getPath().contains(".")) {
117                    if (ed.getMin() == 1) {
118                        if (parentChainHasOptional(ed, sd)) {
119                            requiredNesteds++;
120                        } else {
121                            requiredOutrights++;
122                        }
123                    }
124                    if ("0".equals(ed.getMax())) {
125                        prohibits++;
126                    }
127                    if (ed.getMustSupport()) {
128                        supports++;
129                    }
130                    if (ed.hasFixed()) {
131                        fixeds++;
132                    }
133
134                    for (TypeRefComponent t : ed.getType()) {
135                        if (t.hasProfile() && !igp.isDatatype(t.getProfile().get(0).getValue().substring(40))) {
136                            if (ed.getPath().endsWith(".extension")) {
137                                tryAdd(ext, summariseExtension(t.getProfile(), false));
138                            } else if (ed.getPath().endsWith(".modifierExtension")) {
139                                tryAdd(ext, summariseExtension(t.getProfile(), true));
140                            } else {
141                                tryAdd(refs, describeProfile(t.getProfile().get(0).getValue()));
142                            }
143                        }
144                        for (CanonicalType ct : t.getTargetProfile()) {
145                            tryAdd(refs, describeProfile(ct.getValue()));
146                        }
147                    }
148
149                    if (ed.hasSlicing() && !ed.getPath().endsWith(".extension") && !ed.getPath().endsWith(".modifierExtension")) {
150                        tryAdd(slices, describeSlice(ed.getPath(), ed.getSlicing()));
151                    }
152                }
153            }
154            StringBuilder res = new StringBuilder("<a name=\"summary\"> </a>\r\n<p><b>\r\n" + translate("sd.summary", "Summary") + "\r\n</b></p>\r\n");
155            if (ToolingExtensions.hasExtension(sd, "http://hl7.org/fhir/StructureDefinition/structuredefinition-summary")) {
156                Extension v = ToolingExtensions.getExtension(sd, "http://hl7.org/fhir/StructureDefinition/structuredefinition-summary");
157                res.append(processMarkdown("Profile.summary", (PrimitiveType) v.getValue()));
158            }
159            if (supports + requiredOutrights + requiredNesteds + fixeds + prohibits > 0) {
160                boolean started = false;
161                res.append("<p>");
162                if (requiredOutrights > 0 || requiredNesteds > 0) {
163                    started = true;
164                    res.append(translate("sd.summary", "Mandatory: %s %s", toStr(requiredOutrights), (requiredOutrights > 1 ? translate("sd.summary", Utilities.pluralizeMe("element")) : translate("sd.summary", "element"))));
165                    if (requiredNesteds > 0)
166                        res.append(translate("sd.summary", " (%s nested mandatory %s)", toStr(requiredNesteds), requiredNesteds > 1 ? translate("sd.summary", Utilities.pluralizeMe("element")) : translate("sd.summary", "element")));
167                }
168                if (supports > 0) {
169                    if (started)
170                        res.append("<br/> ");
171                    started = true;
172                    res.append(translate("sd.summary", "Must-Support: %s %s", toStr(supports), supports > 1 ? translate("sd.summary", Utilities.pluralizeMe("element")) : translate("sd.summary", "element")));
173                }
174                if (fixeds > 0) {
175                    if (started)
176                        res.append("<br/> ");
177                    started = true;
178                    res.append(translate("sd.summary", "Fixed Value: %s %s", toStr(fixeds), fixeds > 1 ? translate("sd.summary", Utilities.pluralizeMe("element")) : translate("sd.summary", "element")));
179                }
180                if (prohibits > 0) {
181                    if (started)
182                        res.append("<br/> ");
183                    started = true;
184                    res.append(translate("sd.summary", "Prohibited: %s %s", toStr(prohibits), prohibits > 1 ? translate("sd.summary", Utilities.pluralizeMe("element")) : translate("sd.summary", "element")));
185                }
186                res.append("</p>");
187            }
188
189            if (!refs.isEmpty()) {
190                res.append("<p><b>" + translate("sd.summary", "Structures") + "</b></p>\r\n<p>" + translate("sd.summary", "This structure refers to these other structures") + ":</p>\r\n<ul>\r\n");
191                for (String s : refs)
192                    res.append(s);
193                res.append("\r\n</ul>\r\n\r\n");
194            }
195            if (!ext.isEmpty()) {
196                res.append("<p><b>" + translate("sd.summary", "Extensions") + "</b></p>\r\n<p>" + translate("sd.summary", "This structure refers to these extensions") + ":</p>\r\n<ul>\r\n");
197                for (String s : ext)
198                    res.append(s);
199                res.append("\r\n</ul>\r\n\r\n");
200            }
201            if (!slices.isEmpty()) {
202                res.append("<p><b>" + translate("sd.summary", "Slices") + "</b></p>\r\n<p>" + translate("sd.summary", "This structure defines the following %sSlices%s", "<a href=\"" + corePath + "profiling.html#slices\">", "</a>") + ":</p>\r\n<ul>\r\n");
203                for (String s : slices)
204                    res.append(s);
205                res.append("\r\n</ul>\r\n\r\n");
206            }
207            if (ToolingExtensions.hasExtension(sd, ToolingExtensions.EXT_FMM_LEVEL)) {
208                // Use hard-coded spec link to point to current spec because DSTU2 had maturity listed on a different page
209                res.append("<p><b><a class=\"fmm\" href=\"http://hl7.org/fhir/versions.html#maturity\" title=\"Maturity Level\">" + translate("cs.summary", "Maturity") + "</a></b>: " + ToolingExtensions.readStringExtension(sd, ToolingExtensions.EXT_FMM_LEVEL) + "</p>\r\n");
210            }
211
212            return res.toString();
213        } catch (Exception e) {
214            return "<p><i>" + Utilities.escapeXml(e.getMessage()) + "</i></p>";
215        }
216    }

218    private boolean parentChainHasOptional(ElementDefinition ed, StructureDefinition profile) {
219        if (!ed.getPath().contains("."))
220            return false;
221
222        ElementDefinition match = (ElementDefinition) ed.getUserData(ProfileUtilities.DERIVATION_POINTER);
223        if (match == null)
224            return true; // really, we shouldn't get here, but this appears to be common in the existing profiles?
225        // throw new Error("no matches for "+ed.getPath()+"/"+ed.getName()+" in "+profile.getUrl());
226
227        while (match.getPath().contains(".")) {
228            if (match.getMin() == 0) {
229                return true;
230            }
231            match = getElementParent(profile.getSnapshot().getElement(), match);
232        }
233
234        return false;
235    }

237    private ElementDefinition getElementParent(List<ElementDefinition> list, ElementDefinition element) {
238        String targetPath = element.getPath().substring(0, element.getPath().lastIndexOf("."));
239        int index = list.indexOf(element) - 1;
240        while (index >= 0) {
241            if (list.get(index).getPath().equals(targetPath))
242                return list.get(index);
243            index--;
244        }
245        return null;
246    }

248    private String describeSlice(String path, ElementDefinitionSlicingComponent slicing) {
249        if (!slicing.hasDiscriminator())
250            return "<li>" + translate("sd.summary", "There is a slice with no discriminator at %s", path) + "</li>\r\n";
251        String s = "";
252        if (slicing.getOrdered())
253            s = "ordered";
254        if (slicing.getRules() != SlicingRules.OPEN)
255            s = Utilities.noString(s) ? slicing.getRules().getDisplay() : s + ", " + slicing.getRules().getDisplay();
256        if (!Utilities.noString(s))
257            s = " (" + s + ")";
258        CommaSeparatedStringBuilder b = new CommaSeparatedStringBuilder();
259        for (ElementDefinitionSlicingDiscriminatorComponent d : slicing.getDiscriminator())
260            b.append(d.getType().toCode() + ":" + d.getPath());
261        if (slicing.getDiscriminator().size() == 1)
262            return "<li>" + translate("sd.summary", "The element %s is sliced based on the value of %s", path, b.toString()) + s + "</li>\r\n";
263        else
264            return "<li>" + translate("sd.summary", "The element %s is sliced based on the values of %s", path, b.toString()) + s + "</li>\r\n";
265    }

272    private String summariseExtension(List<CanonicalType> profiles, boolean modifier) throws Exception {
273        if (profiles.size() != 1)
274            throw new Exception("Multiple profiles are not supported at this time (#1)");
275        String url = profiles.get(0).getValue();
276        StructureDefinition ed = context.fetchResource(StructureDefinition.class, url);
277        if (ed == null)
278            return "<li>" + translate("sd.summary", "Unable to summarise extension %s (no extension found)", url) + "</li>";
279        if (ed.getUserData("path") == null)
280            return "<li><a href=\"" + "extension-" + ed.getId().toLowerCase() + ".html\">" + url + "</a>" + (modifier ? " (<b>" + translate("sd.summary", "Modifier") + "</b>) " : "") + "</li>\r\n";
281        else
282            return "<li><a href=\"" + Utilities.escapeXml(ed.getUserString("path")) + "\">" + url + "</a>" + (modifier ? " (<b>" + translate("sd.summary", "Modifier") + "</b>) " : "") + "</li>\r\n";
283    }

295    private String summariseValue(DataType fixed) throws FHIRException {
296        if (fixed instanceof org.hl7.fhir.r5.model.PrimitiveType)
297            return Utilities.escapeXml(((org.hl7.fhir.r5.model.PrimitiveType) fixed).asStringValue());
298        if (fixed instanceof CodeableConcept)
299            return summarise((CodeableConcept) fixed);
300        if (fixed instanceof Coding)
301            return summarise((Coding) fixed);
302        if (fixed instanceof Quantity)
303            return summarise((Quantity) fixed);
304        if (fixed instanceof ContactPoint)
305            return summarise((ContactPoint) fixed);
306        throw new FHIRException("Generating text summary of fixed value not yet done for type " + fixed.getClass().getName());
307    }

358    public String byMustSupport(String defnFile, Set<String> outputTracker) throws IOException, FHIRException, org.hl7.fhir.exceptions.FHIRException {
359        if (sd.getSnapshot().getElement().isEmpty())
360            return "";
361        else {
362            XhtmlComposer composer = new XhtmlComposer(XhtmlComposer.HTML);
363            StructureDefinition sdCopy = sd.copy();
364            Set<ElementDefinition> mustSupport = new HashSet<>();
365
366            List<ElementDefinition> mustSupportElements = new ArrayList<ElementDefinition>();
367
368            // Scan through all the properties checking for must support elements
369            // and clear properties in the cloned StructureDefinition that we don't want to
370            // show in the custom view
371            scanForMustSupport(mustSupport, sdCopy.getSnapshot().getElement(), sdCopy.getSnapshot().getElementFirstRep(), new ArrayList<>());
372            for (ElementDefinition ed : sdCopy.getSnapshot().getElement()) {
373                if (mustSupport.contains(ed)) {
374                    ElementDefinition edCopy = ed.copy();
375                    if (edCopy.hasExample())
376                        edCopy.getExample().clear();
377                    if (!edCopy.getMustSupport()) {
378                        if (edCopy.getPath().contains(".")) {
379                            edCopy.setUserData("render.opaque", true);
380                        }
381                        edCopy.setBinding(null);
382                        ed.getConstraint().clear();
383                    }
384                    edCopy.setMustSupport(false);
385                    mustSupportElements.add(edCopy);
386                }
387            }
388
389            sdCopy.getSnapshot().setElement(mustSupportElements);
390            org.hl7.fhir.utilities.xhtml.XhtmlNode table = utils.generateTable(defnFile, sdCopy, false, destDir, false, sdCopy.getId(), true, corePath, "", false, false, outputTracker, true, true);
391
392            return composer.compose(table);
393        }
394    }

396    private void scanForMustSupport(Set<ElementDefinition> mustSupport, List<ElementDefinition> elements, ElementDefinition element, List<ElementDefinition> parents) {
397        if (parents.isEmpty() || element.hasMustSupport() && element.getMustSupport()) {
398            mustSupport.add(element);
399            for (ElementDefinition parent : parents) {
400                mustSupport.add(parent);
401            }
402        }
403        List<ElementDefinition> children = getChildren(elements, element);
404        for (ElementDefinition child : children) {
405            List<ElementDefinition> np = new ArrayList<>();
406            np.addAll(parents);
407            np.add(element);
408            scanForMustSupport(mustSupport, elements, child, np);
409        }
410    }

419    public String txDiff(boolean withHeadings, boolean mustSupportOnly) throws FHIRException {
420        List<String> txlist = new ArrayList<String>();
421        boolean hasFixed = false;
422        Map<String, ElementDefinition> txmap = new HashMap<String, ElementDefinition>();
423        for (ElementDefinition ed : sd.getDifferential().getElement()) {
424            if (ed.hasBinding() && !"0".equals(ed.getMax()) && (!mustSupportOnly || ed.getMustSupport())) {
425                String id = ed.getId();
426                if (ed.hasFixed()) {
427                    hasFixed = true;
428                    ed.getBinding().setUserData("tx.value", ed.getFixed());
429                } else if (ed.hasPattern()) {
430                    hasFixed = true;
431                    ed.getBinding().setUserData("tx.pattern", ed.getPattern());
432                } else {
433                    // tricky : scan the children for a fixed coding value
434                    DataType t = findFixedValue(ed, true);
435                    if (t != null)
436                        ed.getBinding().setUserData("tx.value", t);
437                }
438                if (ed.getType().size() == 1 && ed.getType().get(0).getWorkingCode().equals("Extension"))
439                    id = id + "<br/>" + ed.getType().get(0).getProfile();
440                txlist.add(id);
441                txmap.put(id, ed);
442            }
443        }
444        if (txlist.isEmpty())
445            return "";
446        else {
447            StringBuilder b = new StringBuilder();
448            if (withHeadings)
449                b.append("<h4>" + translate("sd.tx", "Terminology Bindings (Differential)") + "</h4>\r\n");
450            b.append("<table class=\"list\">\r\n");
451            b.append("<tr><td><b>" + translate("sd.tx", "Path") + "</b></td><td><b>" + translate("sd.tx", "Conformance") + "</b></td><td><b>" + translate("sd.tx", hasFixed ? "ValueSet / Code" : "ValueSet") + "</b></td></tr>\r\n");
452            for (String path : txlist) {
453                txItem(txmap, b, path, sd.getUrl());
454            }
455            b.append("</table>\r\n");
456            return b.toString();
457        }
458    }

709    private void generateAnchors(List<ElementDefinition> stack, Map<String, ElementDefinition> allAnchors) {
710      List<String> list = new ArrayList<>();
711      list.add(stack.get(0).getId()); // initialise
712      for (int i = 1; i < stack.size(); i++) {
713        ElementDefinition ed = stack.get(i);
714        List<String> aliases = new ArrayList<>();
715        String name = tail(ed.getPath());
716        if (name.endsWith("[x]")) {
717          aliases.add(name); 
718          Set<String> tl = new HashSet<String>(); // guard against duplicate type names - can happn in some versions 
719          for (TypeRefComponent tr : ed.getType()) {
720            String tc = tr.getWorkingCode();
721            if (!tl.contains(tc)) {
722              aliases.add(name.replace("[x]", Utilities.capitalize(tc)));
723              aliases.add(name+":"+name.replace("[x]", Utilities.capitalize(tc)));
724              tl.add(tc);
725            }
726          }          
727        } else if (ed.hasSliceName()) {
728          aliases.add(name+":"+ed.getSliceName());    
729          // names.add(name); no good generating this?  
730        } else { 
731          aliases.add(name);
732        }
733        List<String> generated = new ArrayList<>();
734        for (String l : list) {
735          for (String a : aliases) {
736            generated.add(l+"."+a);
737          }
738        }
739        list.clear();
740        list.addAll(generated);
741      }
742      ElementDefinition ed = stack.get(stack.size()-1);
743      
744      // now we have all the possible names, but some of them might be inappropriate if we've 
745      // already generated a type slicer. On the other hand, if we've already done that, we're 
746      // going to steal any type specific ones off it. 
747      List<String> removed = new ArrayList<>();
748      for (String s : list) {
749        if (!allAnchors.containsKey(s)) {
750          allAnchors.put(s, ed);          
751        } else if (s.endsWith("[x]")) {
752          // that belongs on the earlier element
753          removed.add(s);
754        } else {
755          // we delete it from the other 
756          @SuppressWarnings("unchecked")
757          List<String> other = (List<String>) allAnchors.get(s).getUserData("dict.generator.anchors");
758          other.remove(s);
759          allAnchors.put(s, ed);          
760        }
761      }
762      list.removeAll(removed);      
763      ed.setUserData("dict.generator.anchors", list);
764    }

2003    public String references() {
2004        Map<String, String> base = new HashMap<>();
2005        Map<String, String> refs = new HashMap<>();
2006        Map<String, String> trefs = new HashMap<>();
2007        Map<String, String> examples = new HashMap<>();
2008        for (StructureDefinition sd : context.allStructures()) {
2009            if (this.sd.getUrl().equals(sd.getBaseDefinition())) {
2010                base.put(sd.getUserString("path"), sd.present());
2011            }
2012            for (ElementDefinition ed : sd.getSnapshot().getElement()) {
2013                for (TypeRefComponent tr : ed.getType()) {
2014                    for (CanonicalType u : tr.getProfile()) {
2015                        if (this.sd.getUrl().equals(u.getValue())) {
2016                            refs.put(sd.getUserString("path"), sd.present());
2017                        }
2018                    }
2019                    for (CanonicalType u : tr.getTargetProfile()) {
2020                        if (this.sd.getUrl().equals(u.getValue())) {
2021                            trefs.put(sd.getUserString("path"), sd.present());
2022                        }
2023                    }
2024                }
2025            }
2026        }
2027        for (FetchedFile f : files) {
2028            for (FetchedResource r : f.getResources()) {
2029                if (usesSD(r.getElement())) {
2030                    String p = igp.getLinkFor(r, true);
2031                    examples.put(p, r.getTitle());
2032                }
2033            }
2034        }
2035
2036        StringBuilder b = new StringBuilder();
2037        b.append("<p><b>Usage:</b></p>\r\n<ul>\r\n");
2038        if (!base.isEmpty())
2039            b.append(" <li>Derived from this " + sd.describeType() + ": " + refList(base) + "</li>\r\n");
2040        if (!refs.isEmpty())
2041            b.append(" <li>Use this " + sd.describeType() + ": " + refList(refs) + "</li>\r\n");
2042        if (!trefs.isEmpty())
2043            b.append(" <li>Refer to this " + sd.describeType() + ": " + refList(refs) + "</li>\r\n");
2044        if (!examples.isEmpty())
2045            b.append(" <li>Examples for this " + sd.describeType() + ": " + refList(examples) + "</li>\r\n");
2046        if (base.isEmpty() && refs.isEmpty() && trefs.isEmpty() && examples.isEmpty())
2047            b.append(" <li>This " + sd.describeType() + " is not used by any profiles in this Implementation Guide</li>\r\n");
2048        b.append("</ul>\r\n");
2049        return b.toString();
2050    }
