065  public enum ServerType {
066    APACHE, ASP1, ASP2, LITESPEED;
067
068    public static ServerType fromCode(String st) {
069      st = st.toLowerCase();
070      if (st.equals("asp-old"))
071        return ServerType.ASP1;
072      else if (st.equals("asp-new"))
073        return ServerType.ASP2;
074      else if (st.equals("apache"))
075        return ServerType.APACHE;
076      else if (st.equals("litespeed"))
077        return ServerType.LITESPEED;
078      else 
079        throw new Error("-server-type "+st+" not known - use ASP or Apache");
080    }
081  }

092  public IGReleaseUpdater(String folder, String url, String rootFolder, IGRegistryMaintainer reg, ServerType serverType, List<String> otherSpecs, File sft, boolean full) throws IOException {
093    this.folder = folder;
094    this.url = url;
095    this.rootFolder = rootFolder;
096    this.fullUpdate = full;
097    this.sft = sft;
098    if (!"".equals("http://hl7.org/fhir")) { // keep the main spec out of the registry
099      this.reg = reg;
100    }
101    this.serverType = serverType;
102    if (new File(Utilities.path(folder, "publish.ini")).exists()) {
103      IniFile ini = new IniFile(Utilities.path(folder, "publish.ini"));
104      if (ini.getPropertyNames("ig-dirs") != null) {
105        for (String s : ini.getPropertyNames("ig-dirs")) {
106          this.ignoreList.add(Utilities.path(folder, s));
107        }
108      }      
109    }
110    this.ignoreList.addAll(otherSpecs);
111  }

113  public void check(Map<String, IndexMaintainer> indexes)  {
114    List<String> errs = new ArrayList<>(); 
115    try {
116      String f = Utilities.path(folder, "package-list.json");
117      if (!new File(f).exists())
118        errs.add("unable to find package-list.json");
119      else {
120        JsonObject json = JsonTrackingParser.parseJsonFile(f);
121        String canonical = JSONUtil.str(json, "canonical");
122        String packageId = json.get("package-id").getAsString();
123        String realm = Utilities.charCount(packageId, '.') > 1 ? packageId.split("\\.")[2] : null;
124
125        JsonArray list = json.getAsJsonArray("list");
126        JsonObject root = null;
127        System.out.println("Update "+folder+" for "+canonical);
128        for (JsonElement n : list) {
129          JsonObject o = (JsonObject) n;
130          if (!o.has("version"))
131           throw new Error(folder+" has Version without version");
132          if (!JSONUtil.str(o, "version").equals("current")) {
133            if (o.has("current") && o.get("current").getAsBoolean() && o.has("path") && o.get("path").getAsString().startsWith(canonical+"/")) {
134              root = o;
135            }
136          }
137        }
138        boolean save = false;
139        ImplementationGuideEntry rc = reg == null ? null : reg.seeIg(JSONUtil.str(json, "package-id"), canonical, JSONUtil.str(json, "title"), JSONUtil.str(json, "category"));
140        if (!json.has("category")) {
141          if (Utilities.noString(rc.getCategory())) {
142            errs.add(f+" has no category value");            
143          } else {
144            json.addProperty("category", rc.getCategory());
145            save = true;
146          }
147        }
148        boolean hasRelease = false;
149        List<String> folders = new ArrayList<>();
150        for (JsonElement n : list) {
151          JsonObject o = (JsonObject) n;
152          if (JSONUtil.str(o, "version").equals("current")) {
153            if (reg != null) {
154              reg.seeCiBuild(rc, JSONUtil.str(o, "path"), f);
155            }
156          } else {
157            String v = JSONUtil.str(o, "version");
158            if (!o.has("path"))
159              errs.add("version "+v+" has no path'"); 
160            else {
161              String path = JSONUtil.str(o, "path");
162              String vf = Utilities.path(path.replace(url, rootFolder));
163              if (indexes.containsKey(realm)) {
164                indexes.get(realm).seeEntry(packageId, json, o);
165              }
166
167              if (!o.has("sequence")) {
168                throw new Error("No Sequence value for version "+v+" in "+f);
169              }
170              if (!path.endsWith(".html")) {
171                if (!(new File(vf).exists()))
172                  errs.add("version "+v+" path "+vf+" not found (canonical = "+canonical+", path = "+path+")");
173                else {
174                  folders.add(vf);
175                  save = updateStatement(vf, null, ignoreList, json, o, errs, root, canonical, folder, canonical.equals("http://hl7.org/fhir"), false, list) | save;
176                }
177              }
178              if (o.has("current") && o.get("current").getAsBoolean() && o.has("path") && o.get("path").getAsString().startsWith(canonical+"/")) {
179                root = o;
180              }
181              if (reg != null) {
182                if (JSONUtil.str(o, "status").equals("release") || JSONUtil.str(o, "status").equals("trial-use") || JSONUtil.str(o, "status").equals("update")) {
183                  reg.seeRelease(rc, JSONUtil.str(o, "status").equals("update") ? "STU Update" : JSONUtil.str(o, "sequence"), JSONUtil.str(o, "version"), JSONUtil.str(o, "fhirversion", "fhir-version"), JSONUtil.str(o, "path"));
184                  hasRelease = true;
185                } else if (!hasRelease && VersionUtilities.packageForVersion(JSONUtil.str(o, "fhirversion", "fhir-version")) != null)
186                  reg.seeCandidate(rc, JSONUtil.str(o, "sequence")+" "+Utilities.titleize(JSONUtil.str(o, "status")), JSONUtil.str(o, "version"), JSONUtil.str(o, "fhirversion", "fhir-version"), JSONUtil.str(o, "path"));
187              }
188            }
189          }
190        }
191        if (root != null) {
192          updateStatement(folder, folders, ignoreList, json, root, errs, root, canonical, folder, canonical.equals("http://hl7.org/fhir"), true, list);
193        }
194        if (save)
195          TextFile.stringToFile(new GsonBuilder().setPrettyPrinting().create().toJson(json), f, false);
196        File ht = new File(Utilities.path(folder, "history.template"));
197        if (ht.exists()) {
198          scrubApostrophes(json);
199          String jsonv = new GsonBuilder().create().toJson(json);
200          String html = TextFile.fileToString(ht);
201          html = fixParameter(html, "title", json.get("title").getAsString());
202          html = fixParameter(html, "json", jsonv);
203          html = html.replace("assets/", "assets-hist/");
204          html = html.replace("dist/", "dist-hist/");
205          TextFile.stringToFile(html, Utilities.path(folder, "history.html"), false);
206        }
207        ht = new File(Utilities.path(folder, "directory.template"));
208        if (ht.exists()) {
209          scrubApostrophes(json);
210          String jsonv = new GsonBuilder().create().toJson(json);
211          String html = TextFile.fileToString(ht);
212          html = fixParameter(html, "title", json.get("title").getAsString());
213          html = fixParameter(html, "json", jsonv);
214          TextFile.stringToFile(html, Utilities.path(folder, "directory.html"), false);
215        }
216        checkCopyFolderFromRoot(folder, "dist-hist");
217        checkCopyFolderFromRoot(folder, "assets-hist");
218      }
219        
220    } catch (Exception e) {
221      e.printStackTrace();
222      errs.add(e.getMessage());
223    }
224    if (errs.size() == 0)
225      System.out.println(": ok");
226    else {
227      System.out.println("");
228      for (String s : errs) {
229        System.out.println("    "+s);
230      }      
231    }
232  }

234  private String summariseDate(String d) {
235    if (d == null || d.length() < 10) {
236      return "??";
237    }
238    return d.substring(0,7);

241  private String fixParameter(String html, String name, String value) {
242    while (html.contains("[%"+name+"%]")) {
243      html = html.replace("[%"+name+"%]", value);
244    }
245    return html;
246  }

248  private void checkCopyFolderFromRoot(String focus, String name) throws IOException {
249    File f = new File(Utilities.path(focus, name));
250    if (!f.exists() || f.isDirectory()) {
251      if (f.exists()) {
252        Utilities.clearDirectory(f.getAbsolutePath());
253      }
254
255      File src = new File(Utilities.path(rootFolder, name));
256      if (!src.exists()) {
257        System.out.println("History Error: "+src.getAbsolutePath()+" doe not exist");        
258      } else if (!src.isDirectory()) {
259        System.out.println("History Error: "+src.getAbsolutePath()+" is a file, not a directory");
260      } else {
261        FileUtils.copyDirectory(src, f);
262      }
263    } else  {
264      System.out.println("History Error: "+f.getAbsolutePath()+" is a file, not a directory");
265    }

269  private void scrubApostrophes(JsonObject json) {
270    for (Entry<String, JsonElement> p : json.entrySet()) {
271      if (p.getValue().isJsonPrimitive()) {
272        scrubApostrophesInProperty(p);
273      } else if (p.getValue().isJsonObject()) {
274        scrubApostrophes((JsonObject) p.getValue());
275      } else if (p.getValue().isJsonArray()) {
276        int i = 0;
277        for (JsonElement ai : ((JsonArray) p.getValue())) {
278          if (ai.isJsonPrimitive()) {
279            if (ai.getAsString().contains("'"))
280              throw new Error("Don't know how to handle apostrophes in arrays");
281          } else if (ai.isJsonObject()) {
282            scrubApostrophes((JsonObject) ai);
283          } // no arrays containing arrays in package-list.json
284          i++;
285        }
286      }
287    }
288  }

312    if (s.contains("'")) {
313      s = s.replace("'", "`");
314      p.setValue(new JsonPrimitive(s));

324  private boolean updateStatement(String vf, List<String> ignoreList, List<String> ignoreListOuter, JsonObject ig, JsonObject version, List<String> errs, JsonObject root, String canonical, String canonicalPath, boolean isCore, 
325      boolean isCurrent, JsonArray list) throws FileNotFoundException, IOException, FHIRException, ParseException {
326    if (!fullUpdate) {
327      return false;
328    }
329    boolean vc = false;
330    String fragment = genFragment(ig, version, root, canonical, ignoreList != null, isCore);
331    System.out.println("  "+vf+": "+fragment);
332    IGReleaseVersionUpdater igvu = new IGReleaseVersionUpdater(vf, ignoreList, ignoreListOuter, version, folder);
333    igvu.updateStatement(fragment, ignoreList != null ? 0 : 1);
334    System.out.println("  .. "+igvu.getCountTotal()+" files checked, "+igvu.getCountUpdated()+" updated");
335    igvu.checkXmlJsonClones(vf);
336    System.out.println("  .. "+igvu.getClonedTotal()+" clones checked, "+igvu.getClonedCount()+" updated");
337    if (!isCore) {
338      IGPackageChecker pc = new IGPackageChecker(vf, canonical, JSONUtil.str(version, "path"), JSONUtil.str(ig, "package-id"));
339      String fv = JSONUtil.str(version, "fhirversion", "fhirversion");
340      pc.check(JSONUtil.str(version, "version"), JSONUtil.str(ig, "package-id"), fv, 
341          JSONUtil.str(ig, "title"), new SimpleDateFormat("yyyy-MM-dd", new Locale("en", "US")).parse(JSONUtil.str(version, "date")), JSONUtil.str(version, "path"), canonical, getJurisdiction(vf, fv, ig, version));
342    }
343    IGReleaseRedirectionBuilder rb = new IGReleaseRedirectionBuilder(vf, canonical, JSONUtil.str(version, "path"), rootFolder);
344    if (serverType == ServerType.APACHE) {
345      rb.buildApacheRedirections();
346    } else if (serverType == ServerType.ASP2) {
347      rb.buildNewAspRedirections(isCore, isCore && vf.equals(rootFolder));
348    } else if (serverType == ServerType.ASP1) {
349      rb.buildOldAspRedirections();
350    } else if (serverType == ServerType.LITESPEED) {
351      rb.buildLitespeedRedirections();
352    } else if (!canonical.contains("hl7.org/fhir")) {
353      rb.buildApacheRedirections();
354    } else {
355      rb.buildOldAspRedirections();
356    }
357    System.out.println("  .. "+rb.getCountTotal()+" redirections ("+rb.getCountUpdated()+" created/updated)");
358    new DownloadBuilder(vf, canonical, isCurrent ?  canonical: JSONUtil.str(version, "path")).execute();
359    if (!isCurrent && serverType == ServerType.ASP2) {
360      new VersionRedirectorGenerator(canonicalPath).execute(JSONUtil.str(version, "version"), JSONUtil.str(version, "path"));
361    }
362    if (!JSONUtil.has(version, "fhirversion", "fhirversion")) {
363      if (rb.getFhirVersion() == null) {
364        System.out.println("Unable to determine FHIR version for "+vf);
365      } else {
366        version.addProperty("fhirversion", rb.getFhirVersion());
367        vc = true;
368      }
369    }
370    // check links:
371    checkFileExists(vf, "package.tgz");
372    checkFileExists(vf, "qa.html");
373    checkFileExists(vf, isCore ? "fhir-spec.zip" : "full-ig.zip");
374    
375    if (sft != null) {
376      String html = TextFile.fileToString(sft);
377      html = fixParameter(html, "title", JSONUtil.str(ig, "title"));
378      html = fixParameter(html, "id", JSONUtil.str(ig, "package-id"));
379      html = fixParameter(html, "version", isCurrent ? "All Versions" : JSONUtil.str(version, "version"));
380      html = fixParameter(html, "path", isCurrent ? canonicalPath : JSONUtil.str(version, "path"));
381      html = fixParameter(html, "history", isCurrent ? "history.html" : "../history.html");
382      html = fixParameter(html, "search-list", searchLinks(isCurrent, version, canonicalPath, list));
383      html = fixParameter(html, "note", isCurrent ? "this search searches all versions of the "+JSONUtil.str(ig, "title")+", including balloted versions. You can also search specific versions" :
384        "this search searches version "+JSONUtil.str(version, "version")+" of the "+JSONUtil.str(ig, "title")+". You can also search other versions, or all versions at once");
385      html = fixParameter(html, "prefix", "");            
386      TextFile.stringToFile(html, Utilities.path(vf, "searchform.html"), false);          
387    }
388
389    return vc;
390  }

392  private String getJurisdiction(String vf, String fv, JsonObject ig, JsonObject version) throws FHIRFormatError, FHIRException, FileNotFoundException, IOException {
393    String inferred = readJurisdictionFromPackageIg(version.has("package-id") ? version.get("package-id").getAsString() : ig.get("package-id").getAsString());
394        
395    // first, we're going to look in the directory to find a set of IG resources.
396    List<String> igs = findCandidateIgs(vf);
397    // then we choose the most likely one
398    if (igs.size() != 1) {
399      System.out.println("Candidate IG resources = "+igs.toString()+". Using "+inferred);
400      return inferred;
401    }
402    String igf = igs.get(0);
403    ImplementationGuide igr;
404    // then we parse it using the specified version
405    try {
406      igr = loadIg(Utilities.path(vf, igf), fv);
407    } catch (Exception e) {
408      System.out.println("Failed to read IG resource "+Utilities.path(vf, igf)+" ("+fv+"): "+e.getMessage());
409      e.printStackTrace();
410      System.out.println("Using "+inferred);
411      return inferred;
412    }
413    
414    if (igr == null) {
415      System.out.println("Can't load IG resource "+Utilities.path(vf, igf)+" ("+fv+"). Using "+inferred);
416      return inferred;      
417    }
418    // then we see if it has a jurisdiction
419    if (!igr.hasJurisdiction()) {
420      System.out.println("IG resource "+igf+": no Jurisdiction. Using "+inferred);
421      return inferred;
422    }
423    for (CodeableConcept cc : igr.getJurisdiction()) {
424      for (Coding c : cc.getCoding()) {
425        String res = c.getSystem()+"#"+c.getCode();
426        if (inferred != null && !inferred.equals(res)) {
427          System.out.println("IG resource "+igs.toString()+": Jurisdiction mismatch. Found "+res+" but package implies "+inferred);          
428        }
429        return res;
430      }
431    }
432    System.out.println("IG resource "+igs.toString()+": no Jurisdiction found. Using "+inferred);
433    return inferred;
434  }

436  private ImplementationGuide loadIg(String igf, String fv) throws FHIRFormatError, FHIRException, FileNotFoundException, IOException {
437    FileInputStream fs = new FileInputStream(igf);
438    try {
439      if (VersionUtilities.isR2Ver(fv)) {
440        return (ImplementationGuide) VersionConvertorFactory_10_50.convertResource(new org.hl7.fhir.dstu2.formats.XmlParser().parse(fs));
441      }
442      if (VersionUtilities.isR3Ver(fv)) {
443        return (ImplementationGuide) VersionConvertorFactory_30_50.convertResource(new org.hl7.fhir.dstu3.formats.XmlParser().parse(fs), new BaseAdvisor_30_50(false));
444      }
445      if (VersionUtilities.isR4Ver(fv)) {
446        return (ImplementationGuide) VersionConvertorFactory_40_50.convertResource(new org.hl7.fhir.r4.formats.XmlParser().parse(fs));
447      }
448      if (VersionUtilities.isR3Ver(fv)) {
449        return (ImplementationGuide) new org.hl7.fhir.r5.formats.XmlParser().parse(fs);
450      }
451      return null;
452    } finally {
453      fs.close();
454    }
455  }

457  private String readJurisdictionFromPackageIg(String id) {
458    String[] p = id.split("\\.");
459    if (p.length >= 3 && "hl7".equals(p[0]) && "fhir".equals(p[1])) {
460     if ("uv".equals(p[2])) {
461       return "http://unstats.un.org/unsd/methods/m49/m49.htm#001";
462     }
463     if ("us".equals(p[2])) {
464       return "urn:iso:std:iso:3166#US";
465     }
466     return null;
467    }
468    return null;
469  }

471  private List<String> findCandidateIgs(String vf) {
472    List<String> res = new ArrayList<>();
473    for (String f : new File(vf).list()) {
474      if (f.endsWith(".xml") && f.startsWith("ImplementationGuide-")) {
475        res.add(f);
476      }
477    }
478    return res;
479  }

481  private String searchLinks(boolean root, JsonObject focus, String canonical, JsonArray list) {
482    StringBuilder b = new StringBuilder();
483    if (!root) {
484      b.append(" <li><a href=\""+canonical+"/searchform.html\">All Versions</a></li>\r\n");
485    }
486    for (JsonElement n : list) {
487      JsonObject o = (JsonObject) n;
488      if (!JSONUtil.str(o, "version").equals("current")) {
489        String v = JSONUtil.str(o, "version");
490        String path = JSONUtil.str(o, "path");
491        String date = JSONUtil.str(o, "date");
492        if (o == focus && !root) {
493          b.append(" <li>"+JSONUtil.str(o, "sequence")+" "+Utilities.titleize(JSONUtil.str(o, "status"))+" (v"+v+", "+summariseDate(date)+") (this version)</li>\r\n");
494        } else {
495          b.append(" <li><a href=\""+path+"/searchform.html\">"+JSONUtil.str(o, "sequence")+" "+Utilities.titleize(JSONUtil.str(o, "status"))+" (v"+v+", "+summariseDate(date)+")</a></li>\r\n");
496        }
497      }
498    }
499    return b.toString();
500  }

510   * The fragment of HTML this generates has 3 parts 
511   * 
512   * 1. statement of what this is 
513   * 2. reference to current version
514   * 3. referenceto list of published versions
515   * 
516   * @param version
517   * @param root

521  private String genFragment(JsonObject ig, JsonObject version, JsonObject root, String canonical, boolean currentPublication, boolean isCore) {
522    String p1 = JSONUtil.str(ig, "title")+" (v"+JSONUtil.str(version, "version")+": "+state(ig, version)+")";
523    if (!isCore) {
524      p1 = p1 + (version.has("fhirversion") ? " based on <a href=\"http://hl7.org/fhir/"+getPath(JSONUtil.str(version, "fhirversion", "fhir-version"))+"\">FHIR "+fhirRef(JSONUtil.str(version, "fhirversion"))+"</a>" : "")+". ";
525    } else {
526      p1 = p1 + ". ";      
527    }
528    String p2 = root == null ? "" : version == root ? "This is the current published version"+(currentPublication ? "" : " in its permanent home (it will always be available at this URL)") : "The current version which supercedes this version is <a href=\""+(JSONUtil.str(root, "path").startsWith(canonical) ? canonical : JSONUtil.str(root, "path"))+"{{fn}}\">"+JSONUtil.str(root, "version")+"</a>";
529    String p3;
530    if (canonical.equals("http://hl7.org/fhir"))
531      p3 = " For a full list of available versions, see the <a href=\""+canonical+"/directory.html\">Directory of published versions <img src=\"external.png\" style=\"text-align: baseline\"></a>";
532    else
533      p3 = " For a full list of available versions, see the <a href=\""+canonical+"/history.html\">Directory of published versions <img src=\"external.png\" style=\"text-align: baseline\"></a>";
534    return "This page is part of the "+p1+p2+". "+p3;
535  }

597  private String state(JsonObject ig, JsonObject version) {
598    String status = JSONUtil.str(version, "status");
599    String sequence = JSONUtil.str(version, "sequence");
600    if ("trial-use".equals(status))
601      return decorate(sequence);
602    else if ("release".equals(status))
603      return "Release";
604    else if ("qa-preview".equals(status))
605      return "QA Preview";
606    else if ("ballot".equals(status))
607      return decorate(sequence)+" Ballot "+ballotCount(ig, sequence, version);
608    else if ("draft".equals(status))
609      return decorate(sequence)+" Draft";
610    else if ("update".equals(status))
611      return decorate(sequence)+" Update";
612    else if ("normative+trial-use".equals(status))
613      return decorate(sequence+" - Mixed Normative and STU");
614    else if ("normative".equals(status))
615      return decorate(sequence+" - Normative");
616    else 
617      throw new Error("unknown status "+status);
618  }

620  private String decorate(String sequence) {
621    sequence = sequence.replace("Normative", "<a href=\"https://confluence.hl7.org/display/HL7/HL7+Balloting\" title=\"Normative Standard\">Normative</a>");
622    if (sequence.contains("DSTU"))
623      return sequence.replace("DSTU", "<a href=\"https://confluence.hl7.org/display/HL7/HL7+Balloting\" title=\"Draft Standard for Trial-Use\">DSTU</a>");
624    else
625      return sequence.replace("STU", "<a href=\"https://confluence.hl7.org/display/HL7/HL7+Balloting\" title=\"Standard for Trial-Use\">STU</a>");
626  }

628  private String ballotCount(JsonObject ig, String sequence, JsonObject version) {
629    int c = 1;
630    JsonArray list = ig.getAsJsonArray("list");
631    for (int i = list.size() - 1; i >= 0; i--) {
632      JsonObject o = (JsonObject) list.get(i);
633      if (o == version)
634        return Integer.toString(c);
635      if (sequence.equals(JSONUtil.str(o, "sequence")) && "ballot".equals(JSONUtil.str(version, "status")))
636        c++;
637    }
638    return "1";
639  }
