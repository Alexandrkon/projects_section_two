156  public void execute() throws IOException {
157    if (source == null)
158      throw new IOException("A -source parameter is required");
159    if (dest == null)
160      throw new IOException("A -dest parameter is required");
161    if (packageId == null)
162      throw new IOException("A -packageId parameter is required");
163    init();
164    processValidatorPack(new File(source));
165  }
166
167  private void init() throws IOException {
168    pcm = new FilesystemPackageCacheManager(true, ToolsVersion.TOOLS_VERSION);
169    scanner = new Scanner(System. in);
170    paths = new ArrayList<String>();
171  }

173  private void execute(File folder) throws IOException {
174    for (File f : folder.listFiles()) {
175      if (f.isDirectory())
176        execute(f);
177      else if (f.getName().equals("validator.pack")) {
178        processValidatorPack(f);
179      }
180    }
181  }

183  private void processValidatorPack(File f) throws IOException {
184    System.out.println("Processing "+f.getAbsolutePath());
185    try {
186      Map<String, byte[]> files = loadZip(new FileInputStream(f));
187      String version = determineVersion(files);
188      if (Utilities.existsInList(version, "n/a", "3.1.0", "1.8.0")) {
189        System.out.println("  version not supported");
190      } else {
191        ImplementationGuide ig = loadIg(files, version);
192        String canonical = ig.getUrl(); /// ig.getUrl().substring(0, ig.getUrl().indexOf("/ImplementationGuide/"));
193        determinePackageId(ig, canonical, f.getAbsolutePath());
194        checkVersions(ig, version, f.getAbsolutePath());
195        checkLicense(ig);
196
197        System.out.println("  url = "+canonical+", version = "+ig.getVersion()+", fhirversion = "+ig.getFhirVersion()+", id = "+ig.getPackageId()+", license = "+ig.getLicense());
198
199        for (String k : files.keySet()) {
200          if (k.endsWith(".json"))
201            ig.getManifest().addResource().setReference(convertToReference(k));
202        }
203        for (ImplementationGuideDefinitionResourceComponent rd : ig.getDefinition().getResource()) {
204          ManifestResourceComponent ra = getMatchingResource(rd.getReference().getReference(), ig);
205          if (ra != null) {
206            ra.setExample(rd.getExample());
207            if (rd.hasExtension("http://hl7.org/fhir/StructureDefinition/implementationguide-page")) {
208              ra.setRelativePath(rd.getExtensionString("http://hl7.org/fhir/StructureDefinition/implementationguide-page"));
209              rd.removeExtension("http://hl7.org/fhir/StructureDefinition/implementationguide-page");
210            }
211          }
212        }
213
214        if (files.containsKey("spec.internals"))
215          loadSpecInternals(ig, files.get("spec.internals"), version, canonical, files);
216        String destFile = dest != null ? dest : Utilities.path(Utilities.getDirectoryForFile(f.getAbsolutePath()), "package.tgz");
217        String url = Utilities.noString(website) ? canonical : website;
218        NPMPackageGenerator npm = new NPMPackageGenerator(destFile, canonical, url, PackageType.IG, ig, new Date(), false);
219        ByteArrayOutputStream bs = new ByteArrayOutputStream();
220        new JsonParser().setOutputStyle(OutputStyle.NORMAL).compose(bs, ig);
221        npm.addFile(Category.RESOURCE, "ig-r4.json", bs.toByteArray());
222
223        npm.addFile(Category.RESOURCE, "ImplementationGuide-"+ig.getId()+".json", compose(ig, version));
224
225        for (String k : files.keySet()) {
226          if (k.endsWith(".json"))
227            npm.addFile(Category.RESOURCE, k, files.get(k));
228          else if (k.equals("schematron.zip")) {
229            Map<String, byte[]> xfiles = loadZip(new ByteArrayInputStream(files.get(k)));
230            for (String xk : xfiles.keySet())
231              npm.addFile(Category.SCHEMATRON, xk, xfiles.get(xk));
232          } else if (k.equals("spec.internals")) {  // hedging against changes in IG format
233            npm.addFile(Category.OTHER, k, files.get(k));
234          }
235        }
236        npm.finish();
237        System.out.println("  saved to "+npm.filename());
238        paths.add(npm.filename());
239      }
240    } catch (Throwable e) {
241      System.out.println("  error: "+e.getMessage());
242      e.printStackTrace();
243    }
244  }

247  private Reference convertToReference(String k) {
248    k = k.substring(0, k.length()-5);
249    return new Reference(k.substring(0, k.indexOf("-"))+'/'+k.substring(k.indexOf("-")+1));
250  }

252  private byte[] compose(ImplementationGuide ig, String version) throws IOException, FHIRException {
253    if (version.startsWith("1.0")) {
254      return new org.hl7.fhir.dstu2.formats.JsonParser().composeBytes(VersionConvertorFactory_10_50.convertResource(ig));
255    } else if (version.startsWith("1.4")) {
256      return new org.hl7.fhir.dstu2016may.formats.JsonParser().composeBytes(VersionConvertorFactory_14_50.convertResource(ig));
257    } else if (version.startsWith("3.0") ) {
258      return new org.hl7.fhir.dstu3.formats.JsonParser().composeBytes(VersionConvertorFactory_30_50.convertResource(ig));
259    } else if (version.startsWith("4.0") ) {
260      return new org.hl7.fhir.r4.formats.JsonParser().composeBytes(VersionConvertorFactory_40_50.convertResource(ig));
261    } else if (version.equals(Constants.VERSION)) {
262      return new org.hl7.fhir.r5.formats.JsonParser().composeBytes(ig);
263    } else
264      throw new FHIRException("Unsupported version "+version);
265  }

267  private void checkLicense(ImplementationGuide ig) throws IOException, FHIRException {
268    if (!ig.hasLicense()) {
269      if (source != null) {
270        if (license == null)
271          throw new IOException("A -license parameter is required, with a valid SPDX license code, or not-open-source"); 
272        ig.setLicense(SPDXLicense.fromCode(license));
273      }
274    } else
275      ig.setLicense(SPDXLicense.CC01_0);
276    
277  }

279  private void loadSpecInternals(ImplementationGuide ig, byte[] bs, String version, String canonical, Map<String, byte[]> files) throws Exception {
280    SpecMapManager spm = new SpecMapManager(bs, version);
281    ImplementationGuideManifestComponent man = ig.getManifest();
282    man.setRendering(spm.getWebUrl(""));
283    for (String s : spm.getPathUrls()) {
284      if (s.startsWith(canonical)) {
285        String r = s.equals(canonical) ? "" : s.substring(canonical.length() + 1);
286        ManifestResourceComponent ra = getMatchingResource(r, ig);
287        if (ra != null && !ra.hasRelativePath())
288          ra.setRelativePath(spm.getPath(s, null));
289      }
290    }
291    for (String s : spm.getImages()) {
292      ig.getManifest().addImage(s);
293    }
294    for (String s : spm.getTargets()) {
295      if (s.contains("#"))
296        throw new Error("contains # in spec.internal");
297      ig.getManifest().addPage().setName(s);
298    }
299    if (spm.getPages().size() > 0) 
300      throw new Error("contains pages in spec.internal");
301
302  }

304  private ManifestResourceComponent getMatchingResource(String r, ImplementationGuide ig) {
305    for (ManifestResourceComponent t : ig.getManifest().getResource()) 
306      if (r.equals(t.getReference().getReference()))
307        return t;
308    return null;
309  }

311  private void checkVersions(ImplementationGuide ig, String version, String filename) throws FHIRException, IOException {
312    if ("STU3".equals(ig.getFhirVersion()))
313      ig.addFhirVersion(FHIRVersion._3_0_0);
314    
315    if (!ig.hasFhirVersion())
316      ig.addFhirVersion(FHIRVersion.fromCode(version));
317    else if (ig.getFhirVersion().size()>1) {
318      throw new FHIRException("Can't create an IGPack for a multi-version IG");
319    }
320    else {
321      boolean ok = false;
322      for (Enumeration<FHIRVersion> v : ig.getFhirVersion()) {
323        if (version.equals(v.primitiveValue()))
324          ok = true;
325      }
326      if (!ok)
327        throw new FHIRException("FHIR version mismatch: "+version +" vs "+ig.getFhirVersion().get(0));
328    }
329    
330    if (!ig.hasVersion()) {
331      if (packageId != null) {
332        if (versionIg == null)
333          throw new IOException("A -version parameter is required");
334        ig.setVersion(versionIg);
335      } else {
336        String s = tini.getStringProperty("versions", ig.getUrl());
337        while (Utilities.noString(s)) {
338          System.out.print("Enter version for "+ig.getUrl()+": ");
339          s = scanner.nextLine();
340          s = s.trim();
341        }
342        tini.setStringProperty("versions", ig.getUrl(), s, null);
343        ig.setVersion(s);
344      }
345    }
346    
347    for (ImplementationGuideDependsOnComponent d : ig.getDependsOn()) {
348      if (!d.hasVersion()) {
349        if (d.getUri().equals("http://hl7.org/fhir/us/core")) {
350          d.setVersion("1.0.1");
351        }
352      }
353      if (!d.hasPackageId()) {
354        String pid = pcm.getPackageId(d.getUri());
355        boolean post = pid == null;
356        while (Utilities.noString(pid)) {
357          System.out.println("Enter package-id for "+d.getUri()+" from "+filename+":");
358          pid = scanner.nextLine().trim();
359        }
360        d.setPackageId(pid);
361      }
362    }
363  }

365  private void determinePackageId(ImplementationGuide ig, String canonical, String filename) throws IOException, FHIRException {
366    if (packageId != null) {
367      ig.setPackageId(packageId);
368      return;
369    }
370    
371    String pid = pcm.getPackageId(canonical);
372    if (ig.hasPackageId()) {
373      if (!ig.getPackageId().equals(pid))
374        throw new FHIRException("package mismatch "+canonical+"="+ig.getPackageId()+" but cache has "+pid);
375    } else {
376      while (Utilities.noString(pid)) {
377        System.out.println("Enter package-id for "+canonical+" from "+filename+":");
378        pid = scanner.nextLine().trim();
379      }
380      ig.setPackageId(pid);      
381    }
382  }

384  private ImplementationGuide loadIg(Map<String, byte[]> files, String version) throws FHIRException, IOException {
385    String n = null;
386    for (String k : files.keySet()) {
387      if (k.startsWith("ImplementationGuide-"))
388        if (n == null)
389          n = k;
390        else
391          throw new FHIRException("Multiple Implementation Guides found");
392    }
393    if (n == null)
394      throw new FHIRException("Multiple Implementation Guides found");
395    byte[] b = files.get(n);
396    if (version.startsWith("1.0")) {
397      org.hl7.fhir.dstu2.model.Resource r = new org.hl7.fhir.dstu2.formats.JsonParser().parse(b);
398      return (ImplementationGuide) VersionConvertorFactory_10_50.convertResource(r);
399    } else if (version.startsWith("1.4")) {
400      org.hl7.fhir.dstu2016may.model.Resource r = new org.hl7.fhir.dstu2016may.formats.JsonParser().parse(b);
401      return (ImplementationGuide) VersionConvertorFactory_14_50.convertResource(r);
402    } else if (version.startsWith("3.0") ) {
403      org.hl7.fhir.dstu3.model.Resource r = new org.hl7.fhir.dstu3.formats.JsonParser().parse(b);
404      return (ImplementationGuide) VersionConvertorFactory_30_50.convertResource(r);
405    } else if (version.startsWith("4.0") ) {
406      org.hl7.fhir.r4.model.Resource r = new org.hl7.fhir.r4.formats.JsonParser().parse(b);
407      return (ImplementationGuide) VersionConvertorFactory_40_50.convertResource(r);
408    } else if (version.equals(Constants.VERSION)) {
409      org.hl7.fhir.r5.model.Resource r = new org.hl7.fhir.r5.formats.JsonParser().parse(b);
410      return (ImplementationGuide) r;
411    } else
412      throw new FHIRException("Unsupported version "+version);
413  }

415  private String determineVersion(Map<String, byte[]> files) {
416    byte[] b = files.get("version.info");
417    if (b == null)
418      return "n/a";
419    String s = new String(b);
420    s = Utilities.stripBOM(s).trim();
421    while (s.charAt(0) != '[')
422      s = s.substring(1);
423    byte[] bytes = {};
424    try {
425      bytes = s.getBytes("UTF-8");
426    } catch (UnsupportedEncodingException e) {
427
428    }
429    ByteArrayInputStream bs = new ByteArrayInputStream(bytes);
430    IniFile ini = new IniFile(bs);
431    String v = ini.getStringProperty("FHIR", "version");
432    if (v == null)
433      throw new Error("unable to determine version from "+new String(bytes));
434    if (v.startsWith("3.0"))
435      v = "3.0.2";
436    if (v.startsWith("4.0"))
437      v = "4.0.1";
438    return v;
439  }

441  private Map<String, byte[]> loadZip(InputStream stream) throws IOException {
442    Map<String, byte[]> res = new HashMap<String, byte[]>();
443    ZipInputStream zip = new ZipInputStream(stream);
444    ZipEntry ze;
445    while ((ze = zip.getNextEntry()) != null) {
446      int size;
447      byte[] buffer = new byte[2048];
448
449      ByteArrayOutputStream bytes = new ByteArrayOutputStream();
450      BufferedOutputStream bos = new BufferedOutputStream(bytes, buffer.length);
451
452      while ((size = zip.read(buffer, 0, buffer.length)) != -1) {
453        bos.write(buffer, 0, size);
454      }
455      bos.flush();
456      bos.close();
457      res.put(ze.getName(), bytes.toByteArray());
458
459      zip.closeEntry();
460    }
461    zip.close();
462    return res;
463  }
