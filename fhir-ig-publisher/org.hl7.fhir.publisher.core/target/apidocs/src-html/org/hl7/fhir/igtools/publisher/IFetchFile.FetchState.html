034public interface IFetchFile {

039

041  FetchedFile fetchFlexible(String path) throws Exception;

047
048  public ILoggingService getLogger();

050  void setResourceDirs(List<String> theResourceDirs);
051  InputStream openAsStream(String filename) throws FileNotFoundException;
052  String openAsString(String path) throws FileNotFoundException, IOException;
053  void setRootDir(String rootDir);
054}
