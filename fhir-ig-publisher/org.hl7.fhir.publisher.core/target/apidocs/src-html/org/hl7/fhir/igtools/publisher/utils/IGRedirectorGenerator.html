051  private int total;
052
053  /**
054   * 
055   * @param src - the source of the IG to generate redirects for (local folder) 
056   * @param dest - the destination in which to create the redirects (local folder)
057   * @param relativeLocation - the relative path from src to dest - e.g. ../other/something  (yes, it could often be inferred, but not always) Note, this is an HTML path, not a local path 
058   * @throws IOException 
059   */
060  public void generateRedirects(String src, String dest, String relativeLocation) throws IOException {
061    Utilities.clearDirectory(dest);
062    total = 0;
063    processFiles(new File(src), new File(dest), relativeLocation);
064    System.out.println("Created "+total+" redirects for "+src+" in "+dest);
065  }
066
067  private void processFiles(File src, File dest, String relativeLocation) throws IOException {
068    for (File f : src.listFiles()) {
069      if (f.isDirectory()) {
070        File nd = Utilities.createDirectory(Utilities.path(dest.getAbsolutePath(), f.getName()));
071        processFiles(f, nd, "../"+relativeLocation+"/"+f.getName());
072      } else if (f.getName().endsWith(".html")) {
073        genRedirect(dest, f.getName(), relativeLocation);
074      }
075    }
076  }
077
078  private void genRedirect(File dest, String name, String relativeLocation) throws IOException {
079    String redirect = "<!DOCTYPE HTML><html lang=\"en-US\"><head><meta charset=\"UTF-8\">\r\n"+
080        "<meta http-equiv=\"refresh\" content=\"1; url={{url}}\">\r\n"+
081        "<script type=\"text/javascript\">\r\n"+
082        "  window.location.href = \"{{url}}\"\r\n"+
083        "</script>\r\n"+
084        "<title>Page Redirection</title></head><body>If you are not redirected automatically, follow this <a href='{{url}}'>link to Resource Index</a>.</body></html>\r\n";
085    redirect = redirect.replace("{{url}}", relativeLocation+"/"+name);
086    TextFile.stringToFile(redirect, Utilities.path(dest.getAbsolutePath(), name), false);
087    total++;
088  }
