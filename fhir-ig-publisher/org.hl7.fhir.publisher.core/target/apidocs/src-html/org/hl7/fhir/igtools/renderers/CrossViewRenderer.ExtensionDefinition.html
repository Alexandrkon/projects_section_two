097  private void seeObservation(StructureDefinition sd) {
098    ObservationProfile obs = new ObservationProfile();
099    obs.source = sd;
100    int i = 0;
101    String system = null;
102    while (i < sd.getSnapshot().getElement().size()) {
103      ElementDefinition ed = sd.getSnapshot().getElement().get(i);
104      
105      if (ed.getPath().equals("Observation.category") && ed.hasFixed() && ed.getFixed() instanceof CodeableConcept) {
106        obs.category.addAll(((CodeableConcept) ed.getFixed()).getCoding());
107      }
108      if (ed.getPath().equals("Observation.category.coding")) {
109        system = null;
110        if (ed.hasFixed() && ed.getFixed() instanceof Coding) {
111           obs.category.add(((Coding) ed.getFixed()));
112        }
113      }
114      if (ed.getPath().equals("Observation.category.coding.system") && ed.hasFixed()) {
115        system = ed.getFixed().primitiveValue();
116      }
117      if (ed.getPath().equals("Observation.category.coding.code") && (system != null && ed.hasFixed())) {
118        obs.category.add(new Coding(system, ed.getFixed().primitiveValue(), null));
119        system = null;
120      }
121      
122      if (ed.getPath().equals("Observation.code") && ed.hasFixed() && ed.getFixed() instanceof CodeableConcept) {
123        obs.code.addAll(((CodeableConcept) ed.getFixed()).getCoding());
124      }
125      if (ed.getPath().equals("Observation.code.coding")) {
126        system = null;
127        if (ed.hasFixed() && ed.getFixed() instanceof Coding) {
128          obs.code.add(((Coding) ed.getFixed()));
129        }
130      }
131      if (ed.getPath().equals("Observation.code.coding.system") && ed.hasFixed()) {
132        system = ed.getFixed().primitiveValue();
133      }
134      if (ed.getPath().equals("Observation.code.coding.code") && (system != null && ed.hasFixed())) {
135        obs.code.add(new Coding(system, ed.getFixed().primitiveValue(), null));
136        system = null;
137      }
138
139      if (ed.getPath().equals("Observation.bodySite") && ed.hasFixed() && ed.getFixed() instanceof CodeableConcept) {
140        obs.bodySite.addAll(((CodeableConcept) ed.getFixed()).getCoding());
141      }
142      if (ed.getPath().equals("Observation.bodySite.coding")) {
143        system = null;
144        if (ed.hasFixed() && ed.getFixed() instanceof Coding) {
145          obs.bodySite.add(((Coding) ed.getFixed()));
146        }
147      }
148      if (ed.getPath().equals("Observation.bodySite.coding.system") && ed.hasFixed()) {
149        system = ed.getFixed().primitiveValue();
150      }
151      if (ed.getPath().equals("Observation.bodySite.coding.code") && (system != null && ed.hasFixed())) {
152        obs.bodySite.add(new Coding(system, ed.getFixed().primitiveValue(), null));
153        system = null;
154      }
155
156      if (ed.getPath().equals("Observation.method") && ed.hasFixed() && ed.getFixed() instanceof CodeableConcept) {
157        obs.method.addAll(((CodeableConcept) ed.getFixed()).getCoding());
158      }
159      if (ed.getPath().equals("Observation.method.coding")) {
160        system = null;
161        if (ed.hasFixed() && ed.getFixed() instanceof Coding) {
162          obs.method.add(((Coding) ed.getFixed()));
163        }
164      }
165      if (ed.getPath().equals("Observation.method.coding.system") && ed.hasFixed()) {
166        system = ed.getFixed().primitiveValue();
167      }
168      if (ed.getPath().equals("Observation.method.coding.code") && (system != null && ed.hasFixed())) {
169        obs.method.add(new Coding(system, ed.getFixed().primitiveValue(), null));
170        system = null;
171      }
172      
173      if (ed.getPath().equals("Observation.effective[x]")) {
174        for (TypeRefComponent tr : ed.getType())
175          if (!obs.effectiveTypes.contains(tr.getCode()))
176            obs.effectiveTypes.add(tr.getCode());
177      }
178      if (ed.getPath().startsWith("Observation.value") && !ed.getMax().equals("0")) {
179        for (TypeRefComponent tr : ed.getType())
180          if (!obs.types.contains(tr.getCode()))
181            obs.types.add(tr.getCode());
182      }
183      if (ed.getPath().startsWith("Observation.component.")) {
184        i = processObservationComponent(obs, sd.getSnapshot().getElement(), i);
185      } else {
186        i++;
187      }
188    }
189
190    if (obs.hasValue()) {
191      obsList.add(obs);
192    }
193  }

195  private int processObservationComponent(ObservationProfile parent, List<ElementDefinition> list, int i) {
196    ObservationProfile obs = new ObservationProfile();
197    String system = null;
198    while (i < list.size() && list.get(i).getPath().startsWith("Observation.component.")) {

201        obs.category.addAll(((CodeableConcept) ed.getFixed()).getCoding());
202      }

204        system = null;
205        if (ed.hasFixed() && ed.getFixed() instanceof Coding) {
206          obs.category.add(((Coding) ed.getFixed()));
207        }
208      }
209      if (ed.getPath().equals("Observation.component.category.coding.system") && ed.hasFixed()) {
210        system = ed.getFixed().primitiveValue();
211      }
212      if (ed.getPath().equals("Observation.component.category.coding.code") && (system != null && ed.hasFixed())) {
213        obs.method.add(new Coding(system, ed.getFixed().primitiveValue(), null));
214        system = null;
215      }
216      
217      if (ed.getPath().equals("Observation.component.code") && ed.hasFixed() && ed.getFixed() instanceof CodeableConcept) {
218        obs.code.addAll(((CodeableConcept) ed.getFixed()).getCoding());
219      }
220      if (ed.getPath().equals("Observation.component.code.coding")) {
221        system = null;
222        if (ed.hasFixed() && ed.getFixed() instanceof Coding) {
223          obs.code.add(((Coding) ed.getFixed()));
224        }
225      }
226      if (ed.getPath().equals("Observation.component.code.coding.system") && ed.hasFixed()) {

236          if (!obs.types.contains(tr.getCode()))
237            obs.types.add(tr.getCode());
238      }
239      i++;
240    }
241    parent.components.add(obs);
242    return i;

245  private void seeExtensionDefinition(StructureDefinition sd) {
246    if (sd.getUrl().length() < canonical.length()+21) {
247      System.out.println("extension url doesn't follow canonical pattern: "+sd.getUrl()+", so omitted from extension summary");
248      return;
249    }
250    ExtensionDefinition exd = new ExtensionDefinition();
251    extList.add(exd);
252    exd.source = sd;
253    exd.code = sd.getUrl().substring(canonical.length()+21);
254    exd.definition = sd.getDescription();
255    int i = 0;
256    while (i < sd.getSnapshot().getElement().size()) {
257      ElementDefinition ed = sd.getSnapshot().getElement().get(i);
258      if (ed.getPath().startsWith("Extension.value") && !ed.getMax().equals("0")) {
259        for (TypeRefComponent tr : ed.getType())
260          if (!exd.types.contains(tr.getCode()))
261            exd.types.add(tr.getCode());
262      } 
263      if (ed.getPath().startsWith("Extension.extension.")) {
264        i = processExtensionComponent(exd, sd.getSnapshot().getElement(), sd.getSnapshot().getElement().get(i-1).getDefinition(), i);
265      } else {
266        i++;
267      }
268    }
269  }

271  private int processExtensionComponent(ExtensionDefinition parent, List<ElementDefinition> list, String defn, int i) {
272    ExtensionDefinition exd = new ExtensionDefinition();
273    exd.definition = defn;
274    while (i < list.size() && list.get(i).getPath().startsWith("Extension.extension.")) {
275      ElementDefinition ed = list.get(i);
276      if (ed.getPath().equals("Extension.extension.url") && ed.hasFixed()) {
277        exd.code = ed.getFixed().primitiveValue();
278        if (exd.code.startsWith(canonical)) {
279          if (exd.code.length() > canonical.length() + 21) {
280            System.out.println("extension code doesn't follow canonical pattern: "+exd.code);
281          } else { 
282            exd.code = exd.code.substring(canonical.length() + 21);
283          }
284        }
285      }
286      
287      if (ed.getPath().startsWith("Extension.extension.value")) {
288        for (TypeRefComponent tr : ed.getType())
289          if (!exd.types.contains(tr.getCode()))
290            exd.types.add(tr.getCode());
291      }
292      i++;
293    }
294    if (exd.code != null)
295      parent.components.add(exd);
296    return i;
297  }

299  public String getExtensionSummary() {
300    StringBuilder b = new StringBuilder();
301    if (extList.size() == 0) {
302      b.append("<p>No Extensions Defined by this Implementation Guide</p>\r\n");
303    } else {
304      Collections.sort(extList, new ExtListSorter());
305      b.append("<table class=\"grid\">\r\n");
306      b.append(" <tr>");
307      b.append("<td><b>Code</b></td>");
308      b.append("<td><b>Value Types</b></td>");
309      b.append("<td><b>Definition</b></td>");
310      b.append("</tr>\r\n");
311      
312      for (ExtensionDefinition op : extList) {
313        b.append(" <tr>");
314        b.append("<td><a href=\""+op.source.getUserString("path")+"\">"+op.code+"</a></td>");
315        renderTypeCell(b, true, op.types);        
316        b.append("<td>"+Utilities.escapeXml(op.definition)+"</td>");
317        b.append("</tr>\r\n");
318        for (ExtensionDefinition inner : op.components) {
319          b.append(" <tr>");
320          b.append("<td>&nbsp;&nbsp;"+inner.code+"</td>");
321          renderTypeCell(b, true, inner.types);        
322          b.append("<td>"+Utilities.escapeXml(inner.definition)+"</td>");
323          b.append("</tr>\r\n");
324          
325        }
326      }
327      b.append("</table>\r\n");
328    }
329    return b.toString();
330  }

333  public String getObservationSummary() {
334    StringBuilder b = new StringBuilder();
335    if (obsList.size() == 0) {
336      b.append("<p>No Observations Found</p>\r\n");
337    } else {
338      Collections.sort(obsList, new ObsListSorter());
339      b.append("<table class=\"grid\">\r\n");
340      boolean hasCat = false;
341      boolean hasCode = false;
342      boolean hasLoinc = false;
343      boolean hasSnomed = false;
344      boolean hasEffective = false;
345      boolean hasTypes = false;
346      boolean hasBodySite = false;
347      boolean hasMethod = false;
348      for (ObservationProfile op : obsList) {
349        hasCat = hasCat || !op.category.isEmpty();
350        hasLoinc = hasLoinc || hasCode(op.code, "http://loinc.org");
351        hasSnomed = hasSnomed || hasCode(op.code, "http://snomed.info/sct");
352        hasCode = hasCode || hasOtherCode(op.code, "http://loinc.org", "http://snomed.info/sct");
353        hasEffective = hasEffective || !op.effectiveTypes.isEmpty();
354        hasTypes = hasTypes || !op.types.isEmpty();
355        hasBodySite = hasBodySite || !op.bodySite.isEmpty();
356        hasMethod = hasMethod || !op.method.isEmpty();
357      }
358      if (hasCat) {
359        List<Coding> cat = obsList.get(0).category;
360        boolean diff = false;
361        for (ObservationProfile op : obsList) {
362          if (!isSameCodes(cat, op.category)) {
363            diff = true;
364          }
365        }
366        if (!diff) {
367          if (cat.isEmpty()) {
368            b.append("<p>All the observations have the no assigned category</p>\r\n");            
369          } else {
370            b.append("<p>All the observations have the category "+new DataRenderer(context).displayCoding(cat)+"</p>\r\n");
371          }
372          hasCat = false;
373        }
374      }
375      b.append(" <tr>");
376      if (hasCat) b.append("<td><b>Category</b></td>");
377      if (hasLoinc) b.append("<td><b>Loinc</b></td>");
378      if (hasSnomed) b.append("<td><b>Snomed</b></td>");
379      if (hasCode) b.append("<td><b>Code</b></td>");
380      b.append("<td><b>Profile Name</b></td>");
381      if (hasEffective) b.append("<td><b>Time Types</b></td>");
382      if (hasTypes) b.append("<td><b>Value Types</b></td>");
383      if (hasBodySite) b.append("<td><b>Body Site</b></td>");
384      if (hasMethod) b.append("<td><b>Method</b></td>");
385      b.append("</tr>\r\n");
386      
387      for (ObservationProfile op : obsList) {
388        b.append(" <tr>");
389
390        renderCodingCell(b, hasCat, op.category, null, null);
391        renderCodingCell(b, hasLoinc, op.code, "http://loinc.org", "https://loinc.org/");
392        renderCodingCell(b, hasSnomed, op.code, "http://snomed.info/sct", null);
393        renderCodingCell(b, hasCode, op.code, null, null);
394        b.append("<td><a href=\""+op.source.getUserString("path")+"\">"+op.source.present()+"</a></td>");
395        renderTypeCell(b, hasEffective, op.effectiveTypes);
396        renderTypeCell(b, hasTypes, op.types);
397        renderCodingCell(b, hasBodySite, op.bodySite, null, null);
398        renderCodingCell(b, hasMethod, op.method, null, null);
399        
400        b.append("</tr>\r\n");
401      }
402      b.append("</table>\r\n");
403    }
404    return b.toString();
405  }

407  private boolean isSameCodes(List<Coding> l1, List<Coding> l2) {
408    if (l1.size() != l2.size())
409      return false;
410    for (Coding c1 : l1) {
411      boolean found = false;
412      for (Coding c2 : l2) {
413        found = found || (c2.hasSystem() && c2.getSystem().equals(c1.getSystem()) && c2.hasCode() && c2.getCode().equals(c1.getCode()));
414      }
415      if (!found)
416        return false;
417    }
418    return true;
419  }

421  private boolean hasOtherCode(List<Coding> codes, String... exclusions) {
422    boolean found = false;
423    for (Coding c : codes) {
424      if (!Utilities.existsInList(c.getSystem(), exclusions)) {
425        found = true;
426      }
427    }
428    return found;  
429  }

431  private boolean hasCode(List<Coding> codes, String system) {
432    for (Coding c : codes) {
433      if (system.equals(c.getSystem())) {
434        return true;
435      }
436    }
437    return false;
438  }

440  private void renderTypeCell(StringBuilder b, boolean render, List<String> types) {
441    if (render) {
442      b.append("<td>");
443      boolean first = true;
444      for (String t : types) {
445        if (first) first = false; else b.append(" | ");
446        StructureDefinition sd = context.fetchTypeDefinition(t);
447        if (sd != null) {
448          b.append("<a href=\""+sd.getUserString("path")+"\">"+t+"</a>");
449        } else {
450          b.append(t);
451        }
452      }
453      b.append("</td>");
454    }
455    
456  }
457  private void renderCodingCell(StringBuilder b, boolean render, List<Coding> list, String system, String link) {
458    if (render) {
459      b.append("<td>");
460      boolean first = true;
461      for (Coding t : list) {
462        if (system == null || system.equals(t.getSystem())) {
463          if (first) first = false; else b.append(", ");
464          String sys = TerminologyRenderer.describeSystem(t.getSystem());
465          if (sys.equals(t.getSystem()))
466            sys = null;
467          if (sys == null) {
468            CodeSystem cs = context.fetchCodeSystem(t.getSystem());
469            if (cs != null)
470              sys = cs.getTitle();
471          }
472          ValidationResult vr = context.validateCode(ValidationOptions.defaults(), t.getSystem(), t.getVersion(), t.getCode(), null);
473          if (system != null) {
474            if (vr != null & vr.getDisplay() != null) {
475              b.append("<a href=\""+link+t.getCode()+"\" title=\""+vr.getDisplay()+"\">"+t.getCode()+"</a>");
476            } else {
477              b.append("<a href=\""+link+t.getCode()+"\">"+t.getCode()+"</a>");
478            }
479          } else {
480            if (vr != null & vr.getDisplay() != null) {
481              if (Utilities.existsInList(t.getSystem(), "http://loinc.org"))
482                b.append("<span title=\""+t.getSystem()+(sys == null ? "" : " ("+sys+")")+": "+ vr.getDisplay()+"\">"+t.getCode()+" "+vr.getDisplay()+"</span>");           
483              else
484                b.append("<span title=\""+t.getSystem()+(sys == null ? "" : " ("+sys+")")+": "+ vr.getDisplay()+"\">"+t.getCode()+"</span>");           
485            } else {
486              b.append("<span title=\""+t.getSystem()+(sys == null ? "" : " ("+sys+"): ")+"\">"+t.getCode()+"</span>");           
487            }
488          }
489        }
490      }
491      b.append("</td>");
492    }
