024import java.io.File;
025import java.io.FileInputStream;
026import java.io.FileNotFoundException;
027import java.io.FileOutputStream;
028import java.io.IOException;
029
030import org.apache.commons.io.IOUtils;
031
032public class ImageUpdater {
033  private String source;
034  private String filename;
035
036  public static void main(String[] args) throws FileNotFoundException, IOException {
037    ImageUpdater self = new ImageUpdater();
038    self.source = args[0];
039    self.filename = args[2];
040    self.process(new File(args[1]));
041    System.out.println("Done");
042  }
043
044  private void process(File folder) throws FileNotFoundException, IOException {
045    System.out.println("Folder: "+folder.getAbsolutePath());
046    for (File f : folder.listFiles()) {
047      if (f.isDirectory()) {
048        process(f);
049      }
050      if (f.getName().equals(filename)) {
051        System.out.println("  Overwrite "+f.getAbsolutePath()+" with "+source);
052        IOUtils.copy(new FileInputStream(source), new FileOutputStream(f));
053      }
054    }
055  }
056
057}
