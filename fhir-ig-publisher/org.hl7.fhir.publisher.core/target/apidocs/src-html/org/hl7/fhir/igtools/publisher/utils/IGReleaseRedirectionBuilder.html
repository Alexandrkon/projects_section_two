146  public void buildApacheRedirections() throws IOException {    
147    Map<String, String> map = createMap(false);
148    if (map != null) {
149      for (String s : map.keySet()) {
150        String path = Utilities.path(folder, s, "index.php");
151        String p = s.replace("/", "-");
152        String litPath = Utilities.path(folder, p);
153        if (new File(litPath+".xml").exists() && new File(litPath+".json").exists()) 
154          createPhpRedirect(path, map.get(s), Utilities.pathURL(vpath, p));
155      }
156    }
157  }

159  public void buildNewAspRedirections(boolean isCore, boolean isCoreRoot) throws IOException {
160    Map<String, String> map = createMap(isCore);
161    if (map == null) {
162      System.out.println("!!! no map, "+folder);
163    } else {
164      // we need to generate a web config, and the redirectors
165      Set<String> rtl = listResourceTypes(map);
166      if (rtl.isEmpty()) {
167        if (!folder.contains("smart-app-launch") && !folder.contains("davinci-deqm")) {
168          System.out.println("!!! empty map, "+folder);
169        }
170      } else {
171        generateWebConfig(rtl, isCoreRoot);
172        for (String rt : rtl) {
173          generateRedirect(rt, map);
174        }
175        if (isCoreRoot) {
176          for (String s : map.keySet()) {
177            if (!s.contains("/")) {
178              String path = Utilities.path(folder, s, "index.asp");
179              String p = s.replace("/", "-");
180              String litPath = Utilities.path(folder, p)+".html";
181              if (!new File(litPath+".xml").exists() && !new File(litPath+".json").exists()) { 
182                litPath = Utilities.path(folder, tail(map.get(s)));
183              } File file = new File(Utilities.changeFileExt(litPath, ".xml"));
184              if (file.exists() && new File(Utilities.changeFileExt(litPath, ".json")).exists()) {
185                createAspRedirect(path, map.get(s), Utilities.pathURL(vpath, head(file.getName())));
186              }
187            }
188          }
189        }
190      }
191    }
192  }

194  private void generateRedirect(String rt, Map<String, String> map) throws IOException {
195    StringBuilder b = new StringBuilder();
196    countTotal++;
197
198    String root = Utilities.pathURL(canonical, rt);
199    b.append("<%@ language=\"javascript\"%>\r\n" + 
200        "\r\n" + 
201        "<%\r\n" + 
202        "  var s = String(Request.ServerVariables(\"HTTP_ACCEPT\"));\r\n" +
203        "  var id = Request.QueryString(\"id\");\r\n"+
204        "  if (s.indexOf(\"application/json+fhir\") > -1) \r\n" + 
205        "    Response.Redirect(\""+root+"-\"+id+\".json2\");\r\n" + 
206        "  else if (s.indexOf(\"application/fhir+json\") > -1) \r\n" + 
207        "    Response.Redirect(\""+root+"-\"+id+\".json1\");\r\n" + 
208        "  else if (s.indexOf(\"application/xml+fhir\") > -1) \r\n" + 
209        "    Response.Redirect(\""+root+"-\"+id+\".xml2\");\r\n" + 
210        "  else if (s.indexOf(\"application/fhir+xml\") > -1) \r\n" + 
211        "    Response.Redirect(\""+root+"-\"+id+\".xml1\");\r\n" + 
212        "  else if (s.indexOf(\"json\") > -1) \r\n" + 
213        "    Response.Redirect(\""+root+"-\"+id+\".json\");\r\n" + 
214        "  else if (s.indexOf(\"html\") == -1) \r\n" + 
215        "    Response.Redirect(\""+root+"-\"+id+\".xml\");\r\n" );
216    for (String s : map.keySet()) {
217      if (s.startsWith(rt+"/")) {
218        String id = s.substring(rt.length()+1);
219        String link = map.get(s);
220        b.append("  else if (id == \""+id+"\")\r\n" + 
221            "    Response.Redirect(\""+link+"\");\r\n");
222      }
223    }
224    b.append("  else if (id == \"index\")\r\n" + 
225        "    Response.Redirect(\""+root+".html\");\r\n");
226    b.append(     
227        "\r\n" + 
228        "%>\r\n" + 
229        "\r\n" + 
230        "<!DOCTYPE html>\r\n" + 
231        "<html>\r\n" + 
232        "<body>\r\n" + 
233        "Internal Error - unknown id <%= Request.QueryString(\"id\") %> (from "+Utilities.path(localFolder, "cr"+rt.toLowerCase()+".asp")+") .\r\n" + 
234        "</body>\r\n" + 
235        "</html>\r\n");
236
237    String asp = b.toString();
238    File f = new File(Utilities.path(folder, "cr"+rt.toLowerCase()+".asp"));
239    if (f.exists()) {
240      String aspc = TextFile.fileToString(f);
241      if (aspc.equals(asp))
242        return;
243    }
244    countUpdated++;
245    TextFile.stringToFile(b.toString(), f);    
246    
247  }

249  private void generateWebConfig(Set<String> rtl, boolean root) throws IOException {
250    StringBuilder b = new StringBuilder();
251    b.append(root ?  WC_START_ROOT : WC_START_OTHER);
252    countTotal++;
253    for (String rt : rtl) { 
254      if (!Utilities.existsInList(rt, "v2", "v3")) {
255        b.append("        <rule name=\""+rulePrefix()+rt+"\">\n" + 
256            "          <match url=\"^("+rt+")/([A-Za-z0-9\\-\\.]{1,64})\" />\n" + 
257            "          <action type=\"Rewrite\" url=\"cr"+rt.toLowerCase()+".asp?type={R:1}&amp;id={R:2}\" />\n" + 
258            "        </rule>\n" + 
259            "");
260      }
261    }
262    b.append(WC_END);
263    String wc = b.toString();
264    File f = new File(Utilities.path(folder, "web.config"));
265    if (f.exists()) {
266      String wcc = TextFile.fileToString(f);
267      if (wcc.equals(wc))
268        return;
269    }
270    countUpdated++;
271    TextFile.stringToFile(b.toString(),f);    
272  }

274  private String rulePrefix() {
275    if (folder.equals(websiteRootFolder)) {
276      return "";
277    }
278    String t = folder.substring(websiteRootFolder.length()+1);
279    t = t.replace("/", ".").replace("\\", ".");
280    return t+".";
281  }

283  private Set<String> listResourceTypes(Map<String, String> map) {
284    Set<String> res = new HashSet<>();
285    for (String s : map.keySet()) {
286      if (s.contains("/")) {
287        res.add(s.substring(0, s.indexOf("/")));
288      }
289    }
290    return res;
291  }

339  private void createLitespeedRedirect(String path, String urlHtml, String urlSrc) throws IOException {
340    String t = PHP_TEMPLATE;
341    t = t.replace("{{html}}", urlHtml);
342    t = t.replace("{{literal}}", urlSrc);
343    Utilities.createDirectory(Utilities.getDirectoryForFile(path));
344    countTotal++;
345    if (!new File(path).exists() || !TextFile.fileToString(path).equals(t)) {
346      TextFile.stringToFile(t, path, false);
347      countUpdated++;
348    }
349  }

375  private Map<String, String> createMap(boolean isCore) throws IOException {
376    if (new File(Utilities.path(folder, "spec.internals")).exists()) {
377      JsonObject json = null;
378      try {
379        json = JsonTrackingParser.parseJsonFile(Utilities.path(folder, "spec.internals"));
380      } catch (Exception e) {
381        return null;
382      }
383      Map<String, String> res = parseSpecDetails(json);
384      if (isCore) {
385        scanAdditionalFiles(res);      
386      }
387      return res;
388    }
389    File f = new File(Utilities.path(folder, "package.tgz"));
390    if (!f.exists())
391      return null;
392    pkg = NpmPackage.fromPackage(new FileInputStream(f));
393    JsonObject json = null;
394    try {
395      json = JsonTrackingParser.parseJson(pkg.load("other", "spec.internals"));
396    } catch (Exception e) {
397      return null;
398    }
399    Map<String, String> res = parseSpecDetails(json);
400    if (isCore) {
401      scanAdditionalFiles(res);      
402    }
403    return res;
404  }

406  private void scanAdditionalFiles(Map<String, String> res) throws IOException {
407    for (File f : new File(folder).listFiles()) {
408      if (f.getName().endsWith(".json") && !f.getName().endsWith(".canonical.json") && new File(Utilities.changeFileExt(f.getAbsolutePath(), ".xml")).exists() && new File(Utilities.changeFileExt(f.getAbsolutePath(), ".html")).exists()) {
409        JsonObject obj = JsonTrackingParser.parseJson(f);
410        if (obj.has("resourceType") && obj.has("id")) {
411          res.put(JSONUtil.str(obj, "resourceType")+"/"+JSONUtil.str(obj, "id"), Utilities.pathURL(vpath, Utilities.changeFileExt(f.getName(), ".html")));
412        }
413      }
414    }
415    
416  }

418  public Map<String, String> parseSpecDetails(JsonObject json) {
419    Map<String, String> res = new HashMap<>();
420    for (Entry<String, JsonElement> p : json.getAsJsonObject("paths").entrySet()) {
421      String key = p.getKey();
422      if (key.contains("|")) {
423        key = key.substring(0,  key.indexOf("|"));
424      }
425      if (key.length() >= canonical.length()+1 && key.startsWith(canonical)) {
426        String value = p.getValue().getAsString();
427        res.put(key.substring(canonical.length()+1), Utilities.pathURL(vpath, value));
428      } else if (key.contains("/")) {
429        res.put(key, Utilities.pathURL(vpath, p.getValue().getAsString()));        
430      }
431    }
432    return res;
433  }
