071    /**
072     * given a string filename that refers to a json config file, find and load the ig, then
073     * merge everything into the ig and save it, along with a config.yml
074     *
075     * @param pcm
076     * @throws Exception
077     */
078    public void convert(String configFile, IWorkerContext context, FilesystemPackageCacheManager pcm) throws Exception {
079        JsonObject configuration = JsonTrackingParser.parseJsonFile(configFile);
080        if (configuration.has("redirect")) { // redirect to support auto-build for complex projects with IG folder in subdirectory
081            String redirectFile = Utilities.path(Utilities.getDirectoryForFile(configFile), configuration.get("redirect").getAsString());
082            configFile = redirectFile;
083            configuration = JsonTrackingParser.parseJsonFile(redirectFile);
084        }
085        String version = ostr(configuration, "version");
086        if (Utilities.noString(version))
087            version = Constants.VERSION;
088        String rootDir = Utilities.getDirectoryForFile(configFile);
089        if (Utilities.noString(rootDir))
090            rootDir = getCurentDirectory();
091        // We need the root to be expressed as a full path.  getDirectoryForFile will do that in general, but not in Eclipse
092        rootDir = new File(rootDir).getCanonicalPath();
093
094        JsonObject paths = configuration.getAsJsonObject("paths");
095        List<String> resourceDirs = new ArrayList<String>();
096        List<String> pagesDirs = new ArrayList<String>();
097        if (paths != null && paths.get("resources") instanceof JsonArray) {
098            for (JsonElement e : (JsonArray) paths.get("resources"))
099                resourceDirs.add(Utilities.path(rootDir, ((JsonPrimitive) e).getAsString()));
100        } else
101            resourceDirs.add(Utilities.path(rootDir, str(paths, "resources", "resources")));
102        if (paths != null && paths.get("pages") instanceof JsonArray) {
103            for (JsonElement e : (JsonArray) paths.get("pages"))
104                pagesDirs.add(Utilities.path(rootDir, ((JsonPrimitive) e).getAsString()));
105        } else
106            pagesDirs.add(Utilities.path(rootDir, str(paths, "pages", "pages")));
107        String qaDir = Utilities.path(rootDir, str(paths, "qa"));
108        String vsCache = ostr(paths, "txCache");
109
110        String igName = Utilities.path(resourceDirs.get(0), str(configuration, "source", "ig.xml"));
111        ImplementationGuide ig = (ImplementationGuide) parse(igName, version);
112
113        // populating the IG from the config file
114        ig.addFhirVersion(FHIRVersion.fromCode(version));
115        if (configuration.has("fixed-business-version")) {
116            ig.setVersion(configuration.getAsJsonPrimitive("fixed-business-version").getAsString());
117            IGHelper.setParameter(ig.getDefinition(), GuideParameterCode.APPLY.toCode(), "version");
118        }
119        if (configuration.has("license"))
120            ig.setLicense(SPDXLicense.fromCode(configuration.getAsJsonPrimitive("license").getAsString()));
121        if (configuration.has("html-template"))
122            IGHelper.setParameter(ig.getDefinition(), GuideParameterCode.HTMLTEMPLATE.toCode(), configuration.getAsJsonPrimitive("html-template").getAsString());
123        for (String s : resourceDirs)
124            ig.addExtension(ToolingExtensions.EXT_IGP_RESOURCES, new StringType(s));
125        for (String s : pagesDirs)
126            ig.addExtension(ToolingExtensions.EXT_IGP_PAGES, new StringType(s));
127        ig.setPackageId(configuration.get("npm-name").getAsString());
128        if (configuration.has("jurisdiction")) {
129            ig.getJurisdiction().clear();
130            for (String s : configuration.getAsJsonPrimitive("jurisdiction").getAsString().trim().split("\\,")) {
131                CodeableConcept cc = ig.addJurisdiction();
132                Coding c = cc.addCoding();
133                String sc = s.trim();
134                if (Utilities.isInteger(sc))
135                    c.setSystem("http://unstats.un.org/unsd/methods/m49/m49.htm").setCode(sc);
136                else
137                    c.setSystem("urn:iso:std:iso:3166").setCode(sc);
138                ValidationResult vr = context.validateCode(new ValidationOptions(), c, null);
139                if (vr.getDisplay() != null)
140                    c.setDisplay(vr.getDisplay());
141            }
142            IGHelper.setParameter(ig.getDefinition(), "apply", "jurisdiction");
143        }
144        IGHelper.addParameter(ig.getDefinition(), "generate", "xml");
145        IGHelper.addParameter(ig.getDefinition(), "generate", "JSON");
146
147        Parameters p = new Parameters();
148        String sct = str(configuration, "sct-edition", "http://snomed.info/sct/900000000000207008");
149        p.addParameter("system-version", "http://snomed.info/sct|" + sct);
150        p.addParameter("activeOnly", "true".equals(ostr(configuration, "no-inactive-codes")));
151        new JsonParser().compose(new FileOutputStream(Utilities.path(Utilities.getDirectoryForFile(igName), "ig-expansion-parameters.json")), ig);
152        p = new Parameters();
153        p.addParameter("broken-links", "true".equals(ostr(configuration, "broken-links")));
154        p.addParameter("check-aggregation", "true".equals(ostr(configuration, "check-aggregation")));
155        p.addParameter("check-mustSupport", "true".equals(ostr(configuration, "check-mustSupport")));
156        p.addParameter("anyExtensionsAllowed", "true".equals(ostr(configuration, "anyExtensionsAllowed")));
157        p.addParameter("hintAboutNonMustSupport", "true".equals(ostr(configuration, "hintAboutNonMustSupport")));
158        p.addParameter("suppressedWarningFile", ostr(configuration, "suppressedWarningFile"));
159        if (paths.get("extension-domains") instanceof JsonArray) {
160            for (JsonElement e : (JsonArray) paths.get("extension-domains"))
161                p.addParameter("extension-domain", ((JsonPrimitive) e).getAsString());
162        }
163        if (paths.get("special-urls") instanceof JsonArray) {
164            for (JsonElement e : (JsonArray) paths.get("special-urls"))
165                p.addParameter("special-url", ((JsonPrimitive) e).getAsString());
166        }
167        new JsonParser().compose(new FileOutputStream(Utilities.path(Utilities.getDirectoryForFile(igName), "ig-validation-parameters.json")), ig);
168        if (paths.get("spreadsheets") instanceof JsonArray) {
169            for (JsonElement e : (JsonArray) paths.get("spreadsheets"))
170                ig.addExtension(ToolingExtensions.EXT_IGP_SPREADSHEET, new StringType(((JsonPrimitive) e).getAsString()));
171        }
172        if (paths.get("bundles") instanceof JsonArray) {
173            for (JsonElement e : (JsonArray) paths.get("bundles"))
174                ig.addExtension(ToolingExtensions.EXT_IGP_BUNDLE, new StringType(((JsonPrimitive) e).getAsString()));
175        }
176        JsonObject defaults = configuration.getAsJsonObject("defaults");
177        if (defaults != null) {
178            JsonObject any = defaults.getAsJsonObject("any");
179            if (any != null) {
180                if (any.has("xml") && Boolean.parseBoolean(any.get("xml").getAsString()))
181                    IGHelper.setParameter(ig.getDefinition(), "generate", "xml");
182                if (any.has("json") && Boolean.parseBoolean(any.get("json").getAsString()))
183                    IGHelper.setParameter(ig.getDefinition(), "generate", "json");
184                if (any.has("ttl") && Boolean.parseBoolean(any.get("ttl").getAsString()))
185                    IGHelper.setParameter(ig.getDefinition(), "generate", "turtle");
186            }
187        }
188
189        JsonArray deps = configuration.getAsJsonArray("dependencyList");
190        ig.getDependsOn().clear();
191        if (deps != null) {
192            for (JsonElement d : deps) {
193                JsonObject dep = (JsonObject) d;
194                String name = str(dep, "name");
195                String canonical = ostr(dep, "location");
196                String igver = ostr(dep, "version");
197                String packageId = ostr(dep, "package");
198
199                if (Utilities.noString(packageId)) {
200                    packageId = pcm.getPackageId(canonical);
201                }
202                if (Utilities.noString(canonical) && !Utilities.noString(packageId)) {
203                    canonical = pcm.getPackageUrl(packageId);
204                }
205                if (Utilities.noString(canonical)) {
206                    throw new Exception("You must specify a canonical URL for the IG " + name);
207                }
208
209                ImplementationGuideDependsOnComponent igd = ig.addDependsOn();
210                igd.setId(name);
211                igd.setUri(canonical);
212                igd.setPackageId(packageId);
213                igd.setVersion(igver);
214            }
215        }
216        if (configuration.has("resources")) {
217            for (Entry<String, JsonElement> pp : configuration.getAsJsonObject("resources").entrySet()) {
218                if (!pp.getKey().startsWith("_")) {
219                    String s = pp.getKey();
220                    if (pp.getValue().isJsonObject()) {
221                        JsonObject o = (JsonObject) pp.getValue();
222                        JsonElement vb = o.get("base");
223                        JsonElement vd = o.get("defns");
224                        JsonElement vf = o.get("format");
225                        JsonElement vs = o.get("source");
226                        if (vb != null && vd != null && vf != null && vs != null) {
227                            ImplementationGuideDefinitionResourceComponent res = getResource(ig.getDefinition(), s);
228                            if (res == null) {
229                                res = ig.getDefinition().addResource();
230                                res.setReference(new Reference(s));
231                            }
232                          res.addExtension(ToolingExtensions.EXT_IGP_BASE, new StringType(((JsonPrimitive) vb).getAsString()));
233                          res.addExtension(ToolingExtensions.EXT_IGP_DEFNS, new StringType(((JsonPrimitive) vd).getAsString()));
234                          res.addExtension(ToolingExtensions.EXT_IGP_FORMAT, new StringType(((JsonPrimitive) vf).getAsString()));
235                          res.addExtension(ToolingExtensions.EXT_IGP_SOURCE, new StringType(((JsonPrimitive) vs).getAsString()));
236                        }
237                    }
238                }
239            }
240        }
241
242        String newfile = Utilities.path(Utilities.getDirectoryForFile(igName), "ig-new.json");
243        new JsonParser().setOutputStyle(OutputStyle.PRETTY).compose(new FileOutputStream(newfile), ig);
244        newfile = Utilities.path(Utilities.getDirectoryForFile(igName), "ig-new.xml");
245        new XmlParser().setOutputStyle(OutputStyle.PRETTY).compose(new FileOutputStream(newfile), ig);
246        IniFile ini = new IniFile(Utilities.path(rootDir, "ig.ini"));
247        ini.setStringProperty("IG", "ig", newfile.substring(rootDir.length() + 1), null);
248        ini.save();
249    }

252    private ImplementationGuideDefinitionResourceComponent getResource(ImplementationGuideDefinitionComponent configuration, String ref) {
253        for (ImplementationGuideDefinitionResourceComponent res : configuration.getResource()) {
254            if (res.getReference().getReference().equals(ref))
255                return res;
256        }
257        return null;
258    }

260    private Resource parse(String filename, String version) throws Exception {
261        if (version.startsWith("3.0")) {
262            org.hl7.fhir.dstu3.model.Resource res;
263            if (filename.contains("json"))
264                res = new org.hl7.fhir.dstu3.formats.JsonParser().parse(new FileInputStream(filename));
265            else if (filename.contains("xml"))
266                res = new org.hl7.fhir.dstu3.formats.XmlParser().parse(new FileInputStream(filename));
267            else
268                throw new Exception("Unable to determine file type for " + filename);
269            return VersionConvertorFactory_30_50.convertResource(res);
270        } else if (version.startsWith("1.4")) {
271            org.hl7.fhir.dstu2016may.model.Resource res;
272            if (filename.contains("json"))
273                res = new org.hl7.fhir.dstu2016may.formats.JsonParser().parse(new FileInputStream(filename));
274            else if (filename.contains("xml"))
275                res = new org.hl7.fhir.dstu2016may.formats.XmlParser().parse(new FileInputStream(filename));
276            else
277                throw new Exception("Unable to determine file type for " + filename);
278            return VersionConvertorFactory_14_50.convertResource(res);
279        } else if (version.startsWith("1.0")) {
280            org.hl7.fhir.dstu2.model.Resource res;
281            if (filename.contains("json"))
282                res = new org.hl7.fhir.dstu2.formats.JsonParser().parse(new FileInputStream(filename));
283            else if (filename.contains("xml"))
284                res = new org.hl7.fhir.dstu2.formats.XmlParser().parse(new FileInputStream(filename));
285            else
286                throw new Exception("Unable to determine file type for " + filename);
287
288            BaseAdvisor_10_50 advisor = new IGR2ConvertorAdvisor5();
289            return VersionConvertorFactory_10_50.convertResource(res, advisor);
290        } else if (version.startsWith("4.0")) {
291            org.hl7.fhir.r4.model.Resource res;
292            if (filename.contains("json"))
293                res = new org.hl7.fhir.r4.formats.JsonParser().parse(new FileInputStream(filename));
294            else if (filename.contains("xml"))
295                res = new org.hl7.fhir.r4.formats.XmlParser().parse(new FileInputStream(filename));
296            else
297                throw new Exception("Unable to determine file type for " + filename);
298            return VersionConvertorFactory_40_50.convertResource(res);
299        } else if (version.equals(Constants.VERSION)) {
300            if (filename.contains("json"))
301                return new JsonParser().parse(new FileInputStream(filename));
302            else if (filename.contains("xml"))
303                return new XmlParser().parse(new FileInputStream(filename));
304            else
305                throw new Exception("Unable to determine file type for " + filename);
306        } else
307            throw new Exception("Unsupported version " + version + " (current = " + Constants.VERSION + " (" + ToolsVersion.TOOLS_VERSION + ")");
308
309    }

311    private String str(JsonObject obj, String name) throws Exception {
312        if (!obj.has(name))
313            throw new Exception("Property " + name + " not found");
314        if (!(obj.get(name) instanceof JsonPrimitive))
315            throw new Exception("Property " + name + " not a primitive");
316        JsonPrimitive p = (JsonPrimitive) obj.get(name);
317        if (!p.isString())
318            throw new Exception("Property " + name + " not a string");
319        return p.getAsString();
320    }
321
322    private String ostr(JsonObject obj, String name) throws Exception {
323        if (obj == null)
324            return null;
325        if (!obj.has(name))
326            return null;
327        if (!(obj.get(name) instanceof JsonPrimitive))
328            return null;
329        JsonPrimitive p = (JsonPrimitive) obj.get(name);
330        if (!p.isString())
331            return null;
332        return p.getAsString();
333    }
334
335    private String str(JsonObject obj, String name, String defValue) throws Exception {
336        if (obj == null || !obj.has(name))
337            return defValue;
338        if (!(obj.get(name) instanceof JsonPrimitive))
339            throw new Exception("Property " + name + " not a primitive");
340        JsonPrimitive p = (JsonPrimitive) obj.get(name);
341        if (!p.isString())
342            throw new Exception("Property " + name + " not a string");
343        return p.getAsString();
344    }

346    private static String getCurentDirectory() {
347        String currentDirectory;
348        File file = new File(".");
349        currentDirectory = file.getAbsolutePath();
350        return currentDirectory;
351    }
